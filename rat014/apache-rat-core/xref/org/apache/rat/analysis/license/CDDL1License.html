19  package org.apache.rat.analysis.license;
20  
21  import org.apache.rat.analysis.IHeaderMatcher;
22  import org.apache.rat.analysis.RatHeaderAnalysisException;
23  import org.apache.rat.api.Document;
24  import org.apache.rat.api.MetaData;
25  
26  /**
27   * Base CDDL 1.0 license.
28   */
29  public class CDDL1License extends BaseLicense implements IHeaderMatcher {
30  
31      public static final String CDDL1_LICENSE_DEFN
32              = "The contents of this file are subject to the terms of the Common Development\n"
33              + "and Distribution License(\"CDDL\") (the \"License\"). You may not use this file\n"
34              + "except in compliance with the License.\n\n";
35  
36      public static final String CDDL1_LICENSE_DEFN_ILLUMOS_STYLE
37              = "The contents of this file are subject to the terms of the\n"
38              + "Common Development and Distribution License (the \"License\")\n"
39              + "You may not use this file except in compliance with the License.\n";
40  
41      private final FullTextMatchingLicense textMatcherBase;
42      private final FullTextMatchingLicense textMatcherIllumosStyle;
43  
44  
45      public CDDL1License() {
46          super(MetaData.RAT_LICENSE_FAMILY_CATEGORY_DATUM_CDLL1,
47                  MetaData.RAT_LICENSE_FAMILY_NAME_DATUM_CDDL1,
48                  "");
49          textMatcherBase = new FullTextMatchingLicense(MetaData.RAT_LICENSE_FAMILY_CATEGORY_DATUM_CDLL1,
50                  MetaData.RAT_LICENSE_FAMILY_NAME_DATUM_CDDL1,
51                  "", CDDL1_LICENSE_DEFN);
52          textMatcherIllumosStyle = new FullTextMatchingLicense(MetaData.RAT_LICENSE_FAMILY_CATEGORY_DATUM_CDLL1,
53                  MetaData.RAT_LICENSE_FAMILY_NAME_DATUM_CDDL1,
54                  "", CDDL1_LICENSE_DEFN_ILLUMOS_STYLE);
55      }
56  
57      public boolean match(Document subject, String s) throws RatHeaderAnalysisException {
58          return textMatcherBase.match(subject, s) ||
59                  textMatcherIllumosStyle.match(subject, s);
60      }
61  
62      public void reset() {
63          textMatcherBase.reset();
64          textMatcherIllumosStyle.reset();
65      }
66  
67  }
