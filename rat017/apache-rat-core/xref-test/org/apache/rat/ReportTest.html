204     static Stream<Arguments> optionsProvider() {
205 
206         List<Arguments> lst = new ArrayList<>();
207         String[] args;
208         Predicate<ReportConfiguration> test;
209 
210 //        Report.ARCHIVE
211         for (ReportConfiguration.Processing processing : ReportConfiguration.Processing.values()) {
212             args = new String[]{longOpt(Report.ARCHIVE), processing.name().toLowerCase()};
213             test = c -> c.getArchiveProcessing() == processing;
214             lst.add(Arguments.of(args, test));
215         }
216 //        Report.COPYRIGHT
217         args = new String[]{longOpt(Report.COPYRIGHT), "My copyright statement"};
218         test = c -> c.getCopyrightMessage() == null;
219         lst.add(Arguments.of(args, test));
220         args = new String[]{longOpt(Report.COPYRIGHT), "My copyright statement",
221                 "--" + Report.ADD.getOptions().stream().filter(o -> !o.isDeprecated()).findAny().get().getLongOpt()};
222         test = c -> "My copyright statement".equals(c.getCopyrightMessage());
223         lst.add(Arguments.of(args, test));
224 
225 //        Report.DIR;
226 
227 //        Report.DRY_RUN;
228         args = new String[]{"--" + Report.DRY_RUN.getLongOpt()};
229         test = ReportConfiguration::isDryRun;
230         lst.add(Arguments.of(args, test));
231 
232 //        Report.EXCLUDE_CLI;
233         args = new String[]{"--" + Report.EXCLUDE_CLI.getLongOpt(), "*.foo","[A-Z]\\.bar", "justbaz"};
234         test = c -> {
235             FilenameFilter f = c.getFilesToIgnore();
236 
237             assertNotNull(f);
238             assertFalse(f.accept(tempDirectory, "some.foo" ), "some.foo");
239             assertFalse(f.accept(tempDirectory, "B.bar"), "B.bar");
240             assertFalse(f.accept(tempDirectory, "justbaz" ), "justbaz");
241             assertTrue(f.accept(tempDirectory, "notbaz"), "notbaz");
242             return true;
243         };
244         lst.add(Arguments.of(args, test));
245 
246 //        Report.EXCLUDE_FILE_CLI;
247         File outputFile = new File(tempDirectory, "exclude.txt");
248         try (FileWriter fw = new FileWriter(outputFile)) {
249             fw.write("*.foo");
250             fw.write(System.lineSeparator());
251             fw.write("[A-Z]\\.bar");
252             fw.write(System.lineSeparator());
253             fw.write("justbaz");
254             fw.write(System.lineSeparator());
255         } catch (IOException e) {
256             throw new RuntimeException(e);
257         }
258         args = new String[]{"--" + Report.EXCLUDE_FILE_CLI.getLongOpt(), outputFile.getPath()};
259         // same test as above
260         lst.add(Arguments.of(args, test));
261 
262 //        Report.FORCE
263         args = new String[]{longOpt(Report.FORCE)};
264         test = ReportConfiguration::isAddingLicensesForced;
265         lst.add(Arguments.of(args, test.negate()));
266 
267         args = new String[]{longOpt(Report.FORCE),
268                 "--" + Report.ADD.getOptions().stream().filter(o -> !o.isDeprecated()).findAny().get().getLongOpt()};
269         lst.add(Arguments.of(args, test));
270 
271 //        Report.LICENSES -- tested in standalone test: testLicenseOption()
272 
273 //        Report.LIST_LICENSES;
274         for (LicenseSetFactory.LicenseFilter f : LicenseSetFactory.LicenseFilter.values()) {
275             args = new String[]{"--" + Report.LIST_LICENSES.getLongOpt(), f.name().toLowerCase()};
276             test = c -> c.listLicenses() == f;
277             lst.add(Arguments.of(args, test));
278         }
279 
280 //        Report.LIST_FAMILIES
281         for (LicenseSetFactory.LicenseFilter f : LicenseSetFactory.LicenseFilter.values()) {
282             args = new String[]{"--" + Report.LIST_FAMILIES.getLongOpt(), f.name().toLowerCase()};
283             test = c -> c.listFamilies() == f;
284             lst.add(Arguments.of(args, test));
285         }
286 
287 //        Report LOG_LEVEL
288         DefaultLog log = (DefaultLog) DefaultLog.getInstance();
289         for (Log.Level l : Log.Level.values()) {
290             args = new String[]{"--" + Report.LOG_LEVEL.getLongOpt(), l.name().toLowerCase()};
291             test = c -> log.getLevel() == l;
292             lst.add(Arguments.of(args, test));
293         }
294 //        Report.NO_DEFAULTS
295         args = new String[]{"--" + Report.NO_DEFAULTS.getLongOpt()};
296         test = c -> {
297             assertEquals(0, c.getLicenses(LicenseSetFactory.LicenseFilter.ALL).size());
298             return true;
299         };
300         lst.add(Arguments.of(args, test));
301 
302 //        Report.OUT;
303         File outFile = new File( tempDirectory, "outexample");
304         args = new String[] {"--"+Report.OUT.getLongOpt(), outFile.getAbsolutePath()};
305         test = c -> {
306             try (OutputStream os = c.getOutput().get()) {
307                 os.write("Hello world".getBytes());
308             } catch (IOException e) {
309                 throw new RuntimeException(e);
310             }
311             try (BufferedReader reader = new BufferedReader( new InputStreamReader( new FileInputStream(outFile)))) {
312                 return "Hello world".equals(reader.readLine());
313             } catch (IOException e) {
314                 throw new RuntimeException(e);
315             }
316         };
317         lst.add(Arguments.of(args, test));
318 
319 //        Report.SCAN_HIDDEN_DIRECTORIES
320         args = new String[]{"--" + Report.SCAN_HIDDEN_DIRECTORIES.getLongOpt()};
321         test = c -> c.getDirectoriesToIgnore() == FalseFileFilter.FALSE;
322         lst.add(Arguments.of(args, test));
323 
324 //        Report.STYLESHEET_CLI
325         test = c -> c.getStyleSheet() != null;
326         for (String sheet : new String[]{"plain-rat", "missing-headers", "unapproved-licenses"}) {
327             args = new String[]{"--" + Report.STYLESHEET_CLI.getLongOpt(), sheet};
328             lst.add(Arguments.of(args, test));
329         }
330         URL url = ReportTest.class.getResource("MatcherContainerResource.txt");
331         args = new String[]{"--" + Report.STYLESHEET_CLI.getLongOpt(), url.getFile()};
332         lst.add(Arguments.of(args, test));
333 
334 //        Report.XML;
335         args = new String[]{longOpt(Report.XML)};
336         test = ReportConfiguration::isStyleReport;
337         lst.add(Arguments.of(args, test.negate()));
338         return lst.stream();
339     }

378     @ParameterizedTest
379     @MethodSource("exclusionsProvider")
380     public void testParseExclusions(String pattern, List<IOFileFilter> expectedPatterns, List<String> logEntries) {
381         TestingLog log = new TestingLog();
382         Optional<FilenameFilter> filter = Report.parseExclusions(log, Collections.singletonList(pattern));
383         if (expectedPatterns.isEmpty()) {
384             assertThat(filter).isEmpty();
385         } else {
386             assertInstanceOf(NotFileFilter.class, filter.get());
387             String result = filter.toString();
388             for (IOFileFilter expectedFilter : expectedPatterns) {
389                 TextUtils.assertContains(expectedFilter.toString(), result);
390             }
391         }
392         assertEquals(log.isEmpty(), logEntries.isEmpty());
393         for (String logEntry : logEntries) {
394             log.assertContains(logEntry);
395         }
396     }

398     public static Stream<Arguments> exclusionsProvider() {
399         List<Arguments> lst = new ArrayList<>();
400 
401         lst.add(Arguments.of( "", Collections.emptyList(), Collections.singletonList("INFO: Ignored 1 lines in your exclusion files as comments or empty lines.")));
402 
403         lst.add(Arguments.of( "# a comment", Collections.emptyList(), Collections.singletonList("INFO: Ignored 1 lines in your exclusion files as comments or empty lines.")));
404 
405         List<IOFileFilter> expected = new ArrayList<>();
406         String pattern = "hello.world";
407         expected.add(new RegexFileFilter(pattern));
408         expected.add(new NameFileFilter(pattern));
409         lst.add(Arguments.of( pattern, expected, Collections.emptyList()));
410 
411         expected = new ArrayList<>();
412         pattern = "[Hh]ello.[Ww]orld";
413         expected.add(new RegexFileFilter(pattern));
414         expected.add(new NameFileFilter(pattern));
415         lst.add(Arguments.of( pattern, expected, Collections.emptyList()));
416 
417         expected = new ArrayList<>();
418         pattern = "hell*.world";
419         expected.add(new RegexFileFilter(pattern));
420         expected.add(new NameFileFilter(pattern));
421         expected.add(WildcardFileFilter.builder().setWildcards(pattern).get());
422         lst.add(Arguments.of( pattern, expected, Collections.emptyList()));
423 
424         // see RAT-265 for issue
425         expected = new ArrayList<>();
426         pattern = "*.world";
427         expected.add(new NameFileFilter(pattern));
428         expected.add(WildcardFileFilter.builder().setWildcards(pattern).get());
429         lst.add(Arguments.of( pattern, expected, Collections.emptyList()));
430 
431         expected = new ArrayList<>();
432         pattern = "hello.*";
433         expected.add(new NameFileFilter(pattern));
434         expected.add(WildcardFileFilter.builder().setWildcards(pattern).get());
435         lst.add(Arguments.of( pattern, expected, Collections.emptyList()));
436 
437         expected = new ArrayList<>();
438         pattern = "?ello.world";
439         expected.add(new NameFileFilter(pattern));
440         expected.add(WildcardFileFilter.builder().setWildcards(pattern).get());
441         lst.add(Arguments.of( pattern, expected, Collections.emptyList()));
442 
443         expected = new ArrayList<>();
444         pattern = "hell?.world";
445         expected.add(new RegexFileFilter(pattern));
446         expected.add(new NameFileFilter(pattern));
447         expected.add(WildcardFileFilter.builder().setWildcards(pattern).get());
448         lst.add(Arguments.of( pattern, expected, Collections.emptyList()));
449 
450         expected = new ArrayList<>();
451         pattern = "hello.worl?";
452         expected.add(new NameFileFilter(pattern));
453         expected.add(WildcardFileFilter.builder().setWildcards(pattern).get());
454         lst.add(Arguments.of( pattern, expected, Collections.emptyList()));
455 
456         return lst.stream();
457     }
