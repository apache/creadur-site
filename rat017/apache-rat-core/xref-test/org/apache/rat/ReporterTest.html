49      /**
50       * Finds a node via xpath on the document. And then checks family, approval and
51       * type of elements of the node.
52       * 
53       * @param doc The document to check/
54       * @param xpath the XPath instance to use.
55       * @param resource the xpath statement to locate the node.
56       * @param licenseInfo the license info for the node. (may = null)
57       * @param type the type of resource located.
58       * @param hasSample true if a sample from the document should be present.
59       * @throws Exception on XPath error.
60       */
61      private static void checkNode(Document doc, XPath xpath, String resource, LicenseInfo licenseInfo, String type,
62              boolean hasSample) throws Exception {
63          XmlUtils.getNode(doc, xpath, String.format("/rat-report/resource[@name='%s'][@type='%s']", resource, type));
64          if (licenseInfo != null) {
65              XmlUtils.getNode(doc, xpath,
66                      String.format("/rat-report/resource[@name='%s'][@type='%s']/license[@id='%s'][@family='%s']",
67                              resource, type, licenseInfo.id, licenseInfo.family));
68              XmlUtils.getNode(doc, xpath,
69                      String.format("/rat-report/resource[@name='%s'][@type='%s']/license[@id='%s'][@approval='%s']",
70                              resource, type, licenseInfo.id, licenseInfo.approval));
71              if (licenseInfo.hasNotes) {
72                  XmlUtils.getNode(doc, xpath,
73                          String.format("/rat-report/resource[@name='%s'][@type='%s']/license[@id='%s']/notes", resource,
74                                  type, licenseInfo.id));
75              }
76          }
77          if (hasSample) {
78              XmlUtils.getNode(doc, xpath,
79                      String.format("/rat-report/resource[@name='%s'][@type='%s']/sample", resource, type));
80          }
81      }

83      @Test
84      public void xmlReportTest() throws Exception {
85          Defaults defaults = Defaults.builder().build(DefaultLog.INSTANCE);
86          ByteArrayOutputStream out = new ByteArrayOutputStream();
87  
88          final String elementsPath = Resources.getResourceDirectory("elements/Source.java");
89          final ReportConfiguration configuration = new ReportConfiguration(DefaultLog.INSTANCE);
90          configuration.setStyleReport(false);
91          configuration.setFrom(defaults);
92          configuration.setReportable(new DirectoryWalker(new File(elementsPath), configuration.getFilesToIgnore(), HiddenFileFilter.HIDDEN));
93          configuration.setOut(() -> out);
94          new Reporter(configuration).output();
95          Document doc = XmlUtils.toDom(new ByteArrayInputStream(out.toByteArray()));
96  
97          XPath xPath = XPathFactory.newInstance().newXPath();
98  
99          XmlUtils.getNode(doc, xPath, "/rat-report[@timestamp]");
100 
101         LicenseInfo apacheLic = new LicenseInfo("AL", true, false);
102         checkNode(doc, xPath, "src/test/resources/elements/ILoggerFactory.java", new LicenseInfo("MIT", true, false),
103                 "STANDARD", false);
104         checkNode(doc, xPath, "src/test/resources/elements/Image.png", null, "BINARY", false);
105         checkNode(doc, xPath, "src/test/resources/elements/LICENSE", null, "NOTICE", false);
106         checkNode(doc, xPath, "src/test/resources/elements/NOTICE", null, "NOTICE", false);
107         checkNode(doc, xPath, "src/test/resources/elements/Source.java", new LicenseInfo("?????", false, false),
108                 "STANDARD", true);
109         checkNode(doc, xPath, "src/test/resources/elements/Text.txt", apacheLic, "STANDARD", false);
110         checkNode(doc, xPath, "src/test/resources/elements/TextHttps.txt", apacheLic, "STANDARD", false);
111         checkNode(doc, xPath, "src/test/resources/elements/Xml.xml", apacheLic, "STANDARD", false);
112         checkNode(doc, xPath, "src/test/resources/elements/buildr.rb", apacheLic, "STANDARD", false);
113         checkNode(doc, xPath, "src/test/resources/elements/dummy.jar", null, "ARCHIVE", false);
114         checkNode(doc, xPath, "src/test/resources/elements/sub/Empty.txt", new LicenseInfo("?????", false, false),
115                 "STANDARD", false);
116         checkNode(doc, xPath, "src/test/resources/elements/tri.txt", apacheLic, "STANDARD", false);
117         checkNode(doc, xPath, "src/test/resources/elements/tri.txt", new LicenseInfo("BSD-3", true, false), "STANDARD",
118                 false);
119         checkNode(doc, xPath, "src/test/resources/elements/tri.txt", new LicenseInfo("TMF", "BSD-3", true, false),
120                 "STANDARD", false);
121         checkNode(doc, xPath, "src/test/resources/elements/generated.txt", new LicenseInfo("GEN", true, true),
122                 "GENERATED", false);
123         NodeList nodeList = (NodeList) xPath.compile("/rat-report/resource").evaluate(doc, XPathConstants.NODESET);
124         assertEquals(14, nodeList.getLength());
125     }

127     private static final String NL = System.lineSeparator();
128     private static final String PARAGRAPH = "*****************************************************";
129     private static final String HEADER = NL + PARAGRAPH + NL + //
130             "Summary" + NL + //
131             "-------" + NL + //
132             "Generated at: ";
133 
134     private String documentOut(boolean approved, Type type, String name) {
135         return String.format("^\\Q%s%s %s\\E$", approved ? " " : "!", type.name().charAt(0), name);
136     }
137 
138     private String licenseOut(String family, String name) {
139         return licenseOut(family, family, name);
140     }
141 
142     private String licenseOut(String family, String id, String name) {
143         return String.format("\\s+\\Q%s\\E\\s+\\Q%s\\E\\s+\\Q%s\\E$", family, id, name);
144     }

146     @Test
147     public void plainReportTest() throws Exception {
148         Defaults defaults = Defaults.builder().build(DefaultLog.INSTANCE);
149         ByteArrayOutputStream out = new ByteArrayOutputStream();
150 
151         final String elementsPath = Resources.getResourceDirectory("elements/Source.java");
152         final ReportConfiguration configuration = new ReportConfiguration(DefaultLog.INSTANCE);
153         configuration.setFrom(defaults);
154         configuration.setReportable(new DirectoryWalker(new File(elementsPath), configuration.getFilesToIgnore(), HiddenFileFilter.HIDDEN));
155         configuration.setOut(() -> out);
156         new Reporter(configuration).output();
157 
158         out.flush();
159         String document = out.toString();
160 
161         assertTrue(document.startsWith(HEADER), "'Generated at' is not present in " + document);
162 
163         TextUtils.assertPatternInOutput("^Notes: 2$", document);
164         TextUtils.assertPatternInOutput("^Binaries: 2$", document);
165         TextUtils.assertPatternInOutput("^Archives: 1$", document);
166         TextUtils.assertPatternInOutput("^Standards: 8$", document);
167         TextUtils.assertPatternInOutput("^Apache Licensed: 5$", document);
168         TextUtils.assertPatternInOutput("^Generated Documents: 1$", document);
169         TextUtils.assertPatternInOutput("^2 Unknown Licenses$", document);
170         TextUtils.assertPatternInOutput(
171                 "^Files with unapproved licenses:\\s+" //
172                         + "\\Qsrc/test/resources/elements/Source.java\\E\\s+" //
173                         + "\\Qsrc/test/resources/elements/sub/Empty.txt\\E\\s",
174                 document);
175         TextUtils.assertPatternInOutput(documentOut(true, Type.ARCHIVE, "src/test/resources/elements/dummy.jar"),
176                 document);
177         TextUtils.assertPatternInOutput(
178                 documentOut(true, Type.STANDARD, "src/test/resources/elements/ILoggerFactory.java")
179                         + licenseOut("MIT", "The MIT License"),
180                 document);
181         TextUtils.assertPatternInOutput(documentOut(true, Type.BINARY, "src/test/resources/elements/Image.png"),
182                 document);
183         TextUtils.assertPatternInOutput(documentOut(true, Type.NOTICE, "src/test/resources/elements/LICENSE"),
184                 document);
185         TextUtils.assertPatternInOutput(documentOut(true, Type.NOTICE, "src/test/resources/elements/NOTICE"), document);
186         TextUtils.assertPatternInOutput(documentOut(false, Type.STANDARD, "src/test/resources/elements/Source.java")
187                 + licenseOut("?????", "Unknown license (Unapproved)"), document);
188         TextUtils.assertPatternInOutput(documentOut(true, Type.STANDARD, "src/test/resources/elements/Text.txt")
189                 + licenseOut("AL", "Apache License Version 2.0"), document);
190         TextUtils.assertPatternInOutput(documentOut(true, Type.STANDARD, "src/test/resources/elements/Xml.xml")
191                 + licenseOut("AL", "Apache License Version 2.0"), document);
192         TextUtils.assertPatternInOutput(documentOut(true, Type.STANDARD, "src/test/resources/elements/buildr.rb")
193                 + licenseOut("AL", "Apache License Version 2.0"), document);
194         TextUtils.assertPatternInOutput(documentOut(true, Type.STANDARD, "src/test/resources/elements/TextHttps.txt")
195                 + licenseOut("AL", "Apache License Version 2.0"), document);
196         TextUtils.assertPatternInOutput(documentOut(true, Type.STANDARD, "src/test/resources/elements/tri.txt")
197                 + licenseOut("AL", "Apache License Version 2.0") + licenseOut("BSD-3", "BSD 3 clause")
198                 + licenseOut("BSD-3", "TMF", "The Telemanagement Forum License"), document);
199         TextUtils.assertPatternInOutput(documentOut(false, Type.STANDARD, "src/test/resources/elements/sub/Empty.txt")
200                 + licenseOut("?????", "Unknown license (Unapproved)"), document);
201     }

203     @Test
204     public void UnapprovedLicensesReportTest() throws Exception {
205         Defaults defaults = Defaults.builder().build(DefaultLog.INSTANCE);
206         ByteArrayOutputStream out = new ByteArrayOutputStream();
207 
208         final String elementsPath = Resources.getResourceDirectory("elements/Source.java");
209         final ReportConfiguration configuration = new ReportConfiguration(DefaultLog.INSTANCE);
210         configuration.setFrom(defaults);
211         configuration.setReportable(new DirectoryWalker(new File(elementsPath), configuration.getFilesToIgnore(), HiddenFileFilter.HIDDEN));
212         configuration.setOut(() -> out);
213         configuration.setStyleSheet(this.getClass().getResource("/org/apache/rat/unapproved-licenses.xsl"));
214         new Reporter(configuration).output();
215 
216         out.flush();
217         String document = out.toString();
218 
219         assertTrue(document.startsWith("Generated at: "), "'Generated at' is not present in " + document);
220 
221         TextUtils.assertPatternInOutput("\\Qsrc/test/resources/elements/Source.java\\E$", document);
222         TextUtils.assertPatternInOutput("\\Qsrc/test/resources/elements/sub/Empty.txt\\E", document);
223     }

225     private static class LicenseInfo {
226         String id;
227         String family;
228         boolean approval;
229         boolean hasNotes;
230 
231         LicenseInfo(String id, boolean approval, boolean hasNotes) {
232             this(id, id, approval, hasNotes);
233         }
234 
235         LicenseInfo(String id, String family, boolean approval, boolean hasNotes) {
236             this.id = id;
237             this.family = ILicenseFamily.makeCategory(family);
238             this.approval = approval;
239             this.hasNotes = hasNotes;
240         }
