33  import org.junit.jupiter.params.ParameterizedTest;
34  import org.junit.jupiter.params.provider.Arguments;
35  import org.junit.jupiter.params.provider.MethodSource;

37  public class CopyrightMatcherTest {
38      private final static int NAME = 0;
39      private final static int TEXT = 1;
40      // to be added
41      private static String[] prefix = { "Copyright", "Â©", "(C)", "(c)", "&copy;" };
42  
43      private final static String[] D = { "-d", " 1990-1991" };
44      private final static String[] DO = { "-d-o", " 1990-1991 an owner" };
45      private final static String[] OD = { "-o-d", " an owner 1990-1991" };
46      private final static String[] DOS = { "-d-o-s", " 1990 - 1991 an owner" };
47      private final static String[] ODS = { "-o-d-s", " an owner 1990 - 1991" };
48      private final static String[] S = { "-s", " 1990" };
49      private final static String[] O = { "-o", " an owner" };
50      private final static String[] OS = { "-o-s", " an owner 1990" };
51      private final static String[] SO = { "-s-o", " 1990 an owner" };
52  
53      private static final int TOTAL_TESTS = prefix.length * 9;
54  
55      static Arguments startStopOwner = Arguments.of("start-stop-owner", "1990", "1991", "an owner",
56              expandResults(DO, OD, DOS, ODS), expandResults(D, S, O, OS, SO));
57      static Arguments startOwner = Arguments.of("start-owner", "1990", null, "an owner", expandResults(OS, SO, OD, ODS),
58              expandResults(D, DO, DOS, S, O));
59      static Arguments start = Arguments.of("start", "1990", null, null, expandResults(D, DO, DOS, S, SO),
60              expandResults(OD, ODS, O, OS));
61      static Arguments owner = Arguments.of("owner", null, null, "an owner", expandResults(O, OD, ODS, OS),
62              expandResults(DO, DOS, S, D, SO));
63      static Arguments nada = Arguments.of("nada", null, null, null, expandResults(D, DO, DOS, S, SO),
64              expandResults(OD, ODS, O, OS));
65  
66      public static Stream<Arguments> parameterProvider() {
67          return Stream.of(startStopOwner, startOwner, start, owner, nada);
68      }
69  
70      private static String[][] expandResults(String[]... args) {
71          List<String[]> arry = new ArrayList<>();
72          for (String pfx : prefix) {
73              Arrays.stream(args).map(origin -> new String[] { pfx + origin[0], pfx + origin[1] }).forEach(arry::add);
74          }
75          return arry.toArray(new String[arry.size()][2]);
76      }
77  
78      private static void verify(String testName, String[][] pass, String[][] fail) {
79          assertEquals(TOTAL_TESTS, pass.length + fail.length, "Wrong number of pass/fail tests");
80          Set<String> passSet = new HashSet<String>();
81          Arrays.stream(pass).forEach(s -> passSet.add(s[0]));
82          Set<String> failSet = new HashSet<String>();
83          Arrays.stream(fail).forEach(s -> failSet.add(s[0]));
84          for (String s : passSet) {
85              assertFalse(failSet.contains(s),
86                      () -> String.format("%s is in both pass and fail sets for %s", s, testName));
87          }
88      }
89  
90      @ParameterizedTest
91      @MethodSource("parameterProvider")
92      public void testPass(String testName, String start, String stop, String owner, String[][] pass, String[][] fail) {
93          verify(testName, pass, fail);
94          CopyrightMatcher matcher = new CopyrightMatcher(start, stop, owner);
95          for (String[] target : pass) {
96              IHeaders headers = AbstractMatcherTest.makeHeaders(target[TEXT], null);
97              assertTrue(matcher.matches(headers), () -> String.format("%s:%s failed", testName, target[NAME]));
98              matcher.reset();
99          }
100     }
101 
102     @ParameterizedTest
103     @MethodSource("parameterProvider")
104     public void testFail(String testName, String start, String stop, String owner, String[][] pass, String[][] fail) {
105         verify(testName, pass, fail);
106         CopyrightMatcher matcher = new CopyrightMatcher(start, stop, owner);
107         for (String[] target : fail) {
108             IHeaders headers = AbstractMatcherTest.makeHeaders(target[TEXT], null);
109             assertFalse(matcher.matches(headers), String.format("%s:%s passed", testName, target[NAME]));
110             matcher.reset();
111         }
112     }
113 }
