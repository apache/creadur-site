19  package org.apache.rat.analysis.license;
20  
21  import static org.junit.jupiter.api.Assertions.assertEquals;
22  import static org.junit.jupiter.api.Assertions.fail;
23  
24  import java.io.BufferedReader;
25  import java.io.File;
26  
27  import org.apache.rat.analysis.HeaderCheckWorker;
28  import org.apache.rat.analysis.IHeaders;
29  import org.apache.rat.license.ILicense;
30  import org.apache.rat.test.utils.Resources;
31  
32  class DirectoryScanner {
33  
34      @SuppressWarnings("boxing") // OK in test code
35      /**
36       * Get list of files in a directory, and scan for license matches
37       * 
38       * @param directory the directory containing the files
39       * @param matcher the license matcher
40       * @param expected the expected result of the each scan
41       * @throws Exception
42       */
43      public static void testFilesInDir(String directory, ILicense license, boolean expected) throws Exception {
44          final File[] resourceFiles = Resources.getResourceFiles(directory);
45          if (resourceFiles.length == 0) {
46              fail("No files found under " + directory);
47          }
48          for (File f : resourceFiles) {
49              try (BufferedReader br = Resources.getBufferedReader(f)) {
50                  IHeaders headers = HeaderCheckWorker.readHeader(br,
51                          HeaderCheckWorker.DEFAULT_NUMBER_OF_RETAINED_HEADER_LINES);
52                  boolean result = license.matches(headers);
53                  assertEquals(expected, result, f.toString());
54              } finally {
55                  license.reset();
56              }
57          }
58      }
59  }
