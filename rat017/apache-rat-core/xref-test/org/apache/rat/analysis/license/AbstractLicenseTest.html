43  /**
44   * Test to see if short form license information will be recognized correctly.
45   *
46   */
47  abstract public class AbstractLicenseTest {
48      private static int NAME = 0;
49      private static int TEXT = 1;
50  
51      private Defaults defaults;
52      protected MetaData data;
53  
54      protected AbstractLicenseTest() {
55      }
56  
57      @BeforeEach
58      public void setup() {
59          data = new MetaData();
60          defaults = Defaults.builder().build(DefaultLog.INSTANCE);
61      }
62  
63      protected ILicense extractCategory(String famId, String id) {
64          Optional<ILicense> result = LicenseSetFactory.search(famId, id, defaults.getLicenses(LicenseFilter.ALL));
65          if (!result.isPresent()) {
66              fail(String.format("No licenses for id: f:%s l:%s", famId, id));
67          }
68          return result.get();
69      }
70  
71      @ParameterizedTest
72      @MethodSource("parameterProvider")
73      public void testMatchProcessing(String id, String familyPattern, String name, String notes, String[][] targets)
74              throws IOException {
75          ILicense license = extractCategory(familyPattern, id);
76          try {
77              for (String[] target : targets) {
78                  if (processText(license, target[TEXT])) {
79                      data.reportOnLicense(license);
80                      assertEquals(1, data.licenses().count());
81                      assertEquals(license, data.licenses().findFirst().get());
82                  } else {
83                      fail(license + " was not matched by " + target[NAME]);
84                  }
85                  license.reset();
86              }
87          } finally {
88              license.reset();
89          }
90      }
91  
92      private boolean processText(ILicense license, String text) throws IOException {
93          try (BufferedReader in = new BufferedReader(new StringReader(text))) {
94              IHeaders headers = HeaderCheckWorker.readHeader(in,
95                      HeaderCheckWorker.DEFAULT_NUMBER_OF_RETAINED_HEADER_LINES);
96              return license.matches(headers);
97          }
98      }
99  
100     @ParameterizedTest
101     @MethodSource("parameterProvider")
102     public void testEmbeddedStrings(String id, String family, String name, String notes, String[][] targets)
103             throws IOException {
104         String formats[] = { "%s", "now is not the time %s for copyright", "#%s", "##%s", "## %s", "##%s##", "## %s ##",
105                 "/*%s*/", "/* %s */" };
106 
107         ILicense license = extractCategory(family, id);
108         try {
109             for (String[] target : targets) {
110                 for (String fmt : formats) {
111                     boolean found = processText(license, String.format(fmt, target[TEXT]));
112                     license.reset();
113                     assertTrue(found, () -> String.format("%s %s did not match pattern '%s' for target string %s", id,
114                             name, fmt, target[NAME]));
115                 }
116             }
117         } finally {
118             license.reset();
119         }
120     }

122 }
