52      /** The list of comment prefixes that are used to filter comment lines.  */

56      public static final Predicate<String> NOT_MATCH_FILTER = s -> s.startsWith("!");

59      public static final Predicate<String> MATCH_FILTER = s -> !s.startsWith("!");

65      /**
66       * Creates predicate that filters out comment and blank lines.  Leading spaces are removed and
67       * if the line then starts with a commentPrefix string it is considered a comment and will be removed
68       *
69       * @param commentPrefixes the list of comment prefixes
70       * @return the Predicate that returns false for lines that start with commentPrefixes or are empty.
71       */
72      public static Predicate<String> commentFilter(final Iterable<String> commentPrefixes) {
73          return s -> {
74              if (StringUtils.isNotBlank(s)) {
75                  int i = 1;
76                  while (StringUtils.isBlank(s.substring(0, i))) {
77                      i++;
78                  }
79                  String trimmed = i > 0 ? s.substring(i - 1) : s;
80                  for (String prefix : commentPrefixes) {
81                      if (trimmed.startsWith(prefix)) {
82                          return false;
83                      }
84                  }
85                  return true;
86              }
87              return false;
88          };
89      }
90  
91      /**
92       * Creates predicate that filters out comment and blank lines.  Leading spaces are removed and
93       * if the line then starts with a commentPrefix string it is considered a comment and will be removed
94       *
95       * @param commentPrefix the prefix string for comments.
96       * @return the Predicate that returns false for lines that start with commentPrefixes or are empty.
97       */
98      public static Predicate<String> commentFilter(final String commentPrefix) {
99          return s -> {
100             if (StringUtils.isNotBlank(s)) {
101                 int i = 1;
102                 while (StringUtils.isBlank(s.substring(0, i))) {
103                     i++;
104                 }
105                 String trimmed = i > 0 ? s.substring(i - 1) : s;
106                 return !trimmed.startsWith(commentPrefix);
107             }
108             return false;
109         };
110     }

112     /**
113      * Create a path matcher from a FileFilter
114      * @param fileFilter the file filter to convert.
115      * @return a Path matcher.
116      */
117     public static PathMatcher fromFileFilter(final FileFilter fileFilter) {
118         return path -> fileFilter.accept(path.toFile());
119     }

121     /**
122      * Create a FileFilter from a PathMatcher.
123      * @param parent the document name for the parent of the file to be filtered.
124      * @param nameMatcher the path matcher to convert
125      * @return a FileFilter.
126      */
127     public static FileFilter asFileFilter(final DocumentName parent, final DocumentNameMatcher nameMatcher) {
128         return file -> nameMatcher.matches(new DocumentName(file, parent));

131     /**
132      * Creates an iterator of Strings from a file of patterns.
133      * Removes comment lines.
134      * @param patternFile the file to read.
135      * @param commentPrefix the prefix string for comments.
136      * @return the iterable of Strings from the file.
137      */
138     public static ExtendedIterator<String> asIterator(final File patternFile, final String commentPrefix) {
139         return asIterator(patternFile, ExclusionUtils.commentFilter(commentPrefix));
140     }

142     /**
143      * Creates an iterator of Strings from a file of patterns.
144      * Removes comment lines.
145      * @param patternFile the file to read.
146      * @param commentFilters A predicate return true for non-comment lines
147      * @return the iterable of Strings from the file.
148      */
149     public static ExtendedIterator<String> asIterator(final File patternFile, final Predicate<String> commentFilters) {
150         verifyFile(patternFile);
151         Objects.requireNonNull(commentFilters, "commentFilters");
152         try {
153             return ExtendedIterator.create(IOUtils.lineIterator(new FileReader(patternFile))).filter(commentFilters);
154         } catch (FileNotFoundException e) {
155             throw new ConfigurationException(format("%s is not a valid file.", patternFile));
156         }
157     }

159     /**
160      * Creates an iterable of Strings from a file of patterns.
161      * Removes comment lines.
162      * @param patternFile the file to read.
163      * @param commentPrefix the prefix string for comments.
164      * @return the iterable of Strings from the file.
165      */
166     public static Iterable<String> asIterable(final File patternFile, final String commentPrefix)  {
167         return asIterable(patternFile, commentFilter(commentPrefix));
168     }
169 
170     /**
171      * Creates an iterable of Strings from a file of patterns.
172      * Removes comment lines.
173      * @param patternFile the file to read.
174      * @param commentFilters A predicate returning true for non-comment lines
175      * @return the iterable of Strings from the file.
176      */
177     public static Iterable<String> asIterable(final File patternFile, final Predicate<String> commentFilters)  {
178         verifyFile(patternFile);
179         Objects.requireNonNull(commentFilters, "commentFilters");
180         // can not return LineIterator directly as the patternFile will not be closed leading
181         // to a resource leak in some cases.
182         try (FileReader reader = new FileReader(patternFile)) {
183             List<String> result = new ArrayList<>();
184             Iterator<String> iter = new LineIterator(reader) {
185                 @Override
186                 protected boolean isValidLine(final String line) {
187                     return commentFilters.test(line);
188                 }
189             };
190             iter.forEachRemaining(result::add);
191             return result;
192         } catch (IOException e) {
193             throw new ConfigurationException("Unable to read file " + patternFile, e);
194         }
195     }

197     /**
198      * Returns {@code true} if the file name represents a hidden file
199      * @param f the file to check.
200      * @return true if it is the name of a hidden file.
201      */
202     public static boolean isHidden(final File f) {
203         String s = f.getName();
204         return s.startsWith(".") && !(s.equals(".") || s.equals(".."));
205     }

207     private static void verifyFile(final File file) {
208         if (file == null || !file.exists() || !file.isFile()) {
209             throw new ConfigurationException(format("%s is not a valid file.", file));
210         }
211     }

213     /**
214      * Normalize a match pattern to the standard that MatchPattern expects
215      * @param pattern the raw pattern text.
216      * @return the pattern text for MatchPattern.
217      */
218     public static String normalizePattern(final String pattern) {
219         String cleanPattern = pattern.trim();
220 
221         if (cleanPattern.startsWith(SelectorUtils.REGEX_HANDLER_PREFIX)) {
222             if (File.separatorChar == '\\') {
223                 cleanPattern = StringUtils.replace(cleanPattern, "/", "\\\\");
224             } else {
225                 cleanPattern = StringUtils.replace(cleanPattern, "\\\\", "/");
226             }
227         } else {
228             cleanPattern = cleanPattern.replace(File.separatorChar == '/' ? '\\' : '/', File.separatorChar);
229 
230             if (cleanPattern.endsWith(File.separator)) {
231                 cleanPattern += "**";
232             }
233         }
234         return cleanPattern;
235     }

237     /**
238      * Create matching patterns from the collection of strings
239      * @param iterable the collection of strings.
240      * @return The match patterns from the strings.
241      */
242     public static Optional<MatchPatterns> matchPattern(final Iterable<String> iterable) {
243         return iterable == null ? Optional.empty() : matchPattern(iterable.iterator());
244     }
245 
246     /**
247      * Create matching patterns from the iterator of strings
248      * @param iter the iterator of strings.
249      * @return The match patterns from the strings.
250      */
251     public static Optional<MatchPatterns> matchPattern(final Iterator<String> iter) {
252         ExtendedIterator<String> eIter = ExtendedIterator.create(iter).filter(MATCH_FILTER)
253                 .map(ExclusionUtils::normalizePattern);
254         return iter.hasNext() ?  Optional.of(MatchPatterns.from(() -> eIter))
255         : Optional.empty();
256     }

259      * Create not matching patterns from the collection of strings
260      * These are strings that start with "!".

262      * @return The match patterns from the strings.
263      */
264     public static Optional<MatchPatterns> notMatchPattern(final Iterable<String> iterable) {
265         return iterable == null ? Optional.empty() : notMatchPattern(iterable.iterator());
266     }

268     /**
269      * Create not matching patterns from the iterator of strings
270      * These are strings that start with "!".
271      * @param iter the collection of strings.
272      * @return The match patterns from the strings.
273      */
274     public static Optional<MatchPatterns> notMatchPattern(final Iterator<String> iter) {
275         ExtendedIterator<String> eIter =  ExtendedIterator.create(iter).filter(NOT_MATCH_FILTER)
276                 .map(s -> normalizePattern(s.substring(1)));
277         return eIter.hasNext() ? Optional.of(MatchPatterns.from(() -> eIter))
278                 : Optional.empty();
279     }

281     /**
282      * @param filename file name to cleanup.
283      * @return
284      */
285     public static String localizeFileName(final String filename) {
286         String part = filename;
287         boolean regexHandlerPrefix = part.startsWith(SelectorUtils.REGEX_HANDLER_PREFIX);
288         boolean antHandlerPrefix = part.startsWith(SelectorUtils.REGEX_HANDLER_PREFIX);
289 
290         if (regexHandlerPrefix) {
291             part = part.substring(SelectorUtils.REGEX_HANDLER_PREFIX.length(),
292                     part.lastIndexOf(SelectorUtils.PATTERN_HANDLER_SUFFIX));
293         } else if (antHandlerPrefix) {
294             part = part.substring(SelectorUtils.ANT_HANDLER_PREFIX.length(),
295                     part.lastIndexOf(SelectorUtils.PATTERN_HANDLER_SUFFIX));
296         }
297 
298         if (part.startsWith(File.separator)) {
299             if (File.separator.equals("\\")) {
300                 // remove an escaped backslash. Otherwise leave it alone.
301                 if (part.length() > 1 && part.charAt(1) == File.separatorChar) {
302                         // an escaped backslash so remove both.
303                         part = part.substring(2);
304                 }
305             } else {
306                 // normal file separator -- remove it.
307                 part = part.substring(1);
308             }
309         }
310 
311         if (regexHandlerPrefix || antHandlerPrefix) {
312             part = format("%s%s%s",
313                     regexHandlerPrefix ? SelectorUtils.REGEX_HANDLER_PREFIX : SelectorUtils.ANT_HANDLER_PREFIX,
314                     part, SelectorUtils.PATTERN_HANDLER_SUFFIX);
315         }
316         return part;
317     }
