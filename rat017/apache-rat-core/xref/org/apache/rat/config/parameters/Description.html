38  /**
39   * A description of a component.
40   */
41  public class Description {
42      /** The type of component this describes */
43      private final ComponentType type;
44      /**
45       * The common name for the component. Set by ConfigComponent.name() or
46       * class/field name.
47       */
48      private final String name;
49      /** The description for the component */
50      private final String desc;
51      /** The class of the getter/setter parameter */
52      private final Class<?> childClass;
53      /** True if the getter/setter expects a collection of childClass objects */
54      private final boolean isCollection;
55      /** True if this component is required. */
56      private final boolean required;
57      /**
58       * A map of name to Description for all the components that are children of the
59       * described component.
60       */
61      private final Map<String, Description> children;
62  
63      /**
64       * A predicate to filter Descriptions by component type.
65       * @param type The component type required in the filter.
66       * @return a predicate that will only return Descriptions with the specified component type.
67       */
68      public static Predicate<Description> typePredicate(final ComponentType type) {
69          return d -> d.getType() == type;
70      }

72      /**
73       * Constructor.
74       * @param type the type of the component.
75       * @param name the name of the component.
76       * @param desc the description of the component.
77       * @param isCollection true if the getter/setter expects a collection
78       * @param childClass the class for expected for the getter/setter.
79       * @param children the collection of descriptions for all the components that
80       * are children of the described component.
81       * @param required If {@code true} the component is required.
82       */
83      public Description(final ComponentType type, final String name, final String desc, final boolean isCollection,
84                         final Class<?> childClass, final Collection<Description> children, final boolean required) {
85          this.type = type;
86          this.name = name;
87          this.desc = desc;
88          this.isCollection = isCollection;
89          this.required = required;
90          if (type == ComponentType.BUILD_PARAMETER) {
91              Method m;
92              try {
93                  m = BuilderParams.class.getMethod(name);
94              } catch (NoSuchMethodException | SecurityException e) {
95                  throw new ConfigurationException(format("'%s' is not a valid BuildParams method", name));
96              }
97              this.childClass = m.getReturnType();
98          } else {
99              this.childClass = childClass;
100         }
101         this.children = new TreeMap<>();
102         if (children != null) {
103             children.forEach(d -> {
104                 this.children.put(d.name, d);
105             });
106         }
107     }

173     /**
174      * Retrieve the value of a the described parameter from the specified object.
175      * If the parameter is a collection return {@code null}.
176      * @param object the object that contains the value.
177      * @return the string value.
178      */
179     public String getParamValue(final Object object) {
180         if (isCollection) {
181             return null;
182         }
183         try {
184             Object val = getter(object.getClass()).invoke(object);
185             return val == null ? null : val.toString();
186         } catch (IllegalArgumentException | IllegalAccessException | InvocationTargetException | NoSuchMethodException
187                 | SecurityException e) {
188             DefaultLog.getInstance().error(format("Can not retrieve value for '%s' from %s%n", name, object.getClass().getName()),
189                     e);
190             return null;
191         }
192     }

194     /**
195      * Gets a map of the parameters that the object contains. For example Copyright
196      * has 'start', 'stop', and 'owner' parameters. Some IHeaderMatchers have simple

198      * parameter (empty string) with the text value.
199      * @return the map of parameters to the objects that represent them.

220         return children.values().stream().filter(filter).collect(Collectors.toList());

232     /**
233      * Returns the getter for the component in the specified class.
234      * @param clazz the Class to get the getter from.
235      * @return the getter Method.
236      * @throws NoSuchMethodException if the class does not have the getter.
237      * @throws SecurityException if the getter can not be accessed.
238      */
239     public Method getter(final Class<?> clazz) throws NoSuchMethodException, SecurityException {
240         return clazz.getMethod(methodName("get"));
241     }
242 
243     /**
244      * Returns the setter for the component in the specified class. Notes:
245      * <ul>
246      * <li>License can not be set in components. They are top level components.</li>
247      * <li>Matcher expects an "add" method that accepts an
248      * IHeaderMatcher.Builder.</li>
249      * <li>Parameter expects a {@code set(String)} method.</li>
250      * <li>Unlabeled expects a {@code set(String)} method.</li>
251      * <li>BuilderParam expects a {@code set} method that takes a
252      * {@code childeClass} argument.</li>
253      * </ul>
254      * @param clazz the Class to get the getter from, generally a Builder class..
255      * @return the getter Method.
256      * @throws NoSuchMethodException if the class does not have the getter.
257      * @throws SecurityException if the getter can not be accessed.
258      */
259     public Method setter(final Class<?> clazz) throws NoSuchMethodException, SecurityException {
260         String methodName = methodName(isCollection ? "add" : "set");
261         switch (type) {
262         case LICENSE:
263             throw new NoSuchMethodException("Can not set a License as a child");
264         case MATCHER:
265             return clazz.getMethod(methodName, IHeaderMatcher.Builder.class);
266         case PARAMETER:
267             return clazz.getMethod(methodName,
268                     IHeaderMatcher.class.isAssignableFrom(childClass) ? IHeaderMatcher.Builder.class : childClass);
269         case BUILD_PARAMETER:
270             return clazz.getMethod(methodName, childClass);
271         }
272         // should not happen
273         throw new IllegalStateException("Type " + type + " not valid.");
274     }
275 
276     private void callSetter(final Description description, final IHeaderMatcher.Builder builder, final String value) {
277         try {
278             description.setter(builder.getClass()).invoke(builder, value);
279         } catch (NoSuchMethodException e) {
280             String msg = format("No setter for '%s' on %s", description.getCommonName(),
281                     builder.getClass().getCanonicalName());
282             DefaultLog.getInstance().error(msg);
283             throw new ConfigurationException(msg);
284         } catch (IllegalAccessException | IllegalArgumentException | InvocationTargetException | SecurityException e) {
285             String msg = format("Unable to call setter for '%s' on %s", description.getCommonName(),
286                     builder.getClass().getCanonicalName());
287             DefaultLog.getInstance().error(msg, e);
288             throw new ConfigurationException(msg, e);
289         }
290     }

292     /**
293      * Sets the children of values in the builder. Sets the parameters to the values
294      * specified in the map. Only children that accept string arguments should be
295      * specified.
296      * @param builder The Matcher builder to set the values in.
297      * @param attributes a Map of parameter names to values.
298      */
299     public void setChildren(final IHeaderMatcher.Builder builder, final Map<String, String> attributes) {
300         attributes.entrySet().forEach(entry -> setChild(builder, entry.getKey(), entry.getValue()));
301     }
302 
303     /**
304      * Sets the child value in the builder.
305      * @param builder The Matcher builder to set the values in.
306      * @param name the name of the child to set
307      * @param value the value of the parameter.
308      */
309     public void setChild(final IHeaderMatcher.Builder builder, final String name, final String value) {
310         Description d = getChildren().get(name);
311         if (d == null) {
312             DefaultLog.getInstance().error(format("%s does not define a ConfigComponent for a member %s.",
313                     builder.getClass().getCanonicalName(), name));
314         } else {
315             callSetter(d, builder, value);
316         }
317     }

319     @Override
320     public String toString() {
321         String childList = children.isEmpty() ? ""
322                 : String.join(", ",
323                         children.values().stream().map(Description::getCommonName).collect(Collectors.toList()));
324 
325         return format("Description[%s t:%s c:%s %s children: [%s]] ", name, type, isCollection, childClass,
326                 childList);
327     }

329     /**
330      * Write a description with indentation.
331      * @param indent the number of spaces to indent.
332      * @return the string with the formatted data.
333      */
334     public String toString(final int indent) {
335         char[] spaces = new char[indent];
336         Arrays.fill(spaces, ' ');
337         String padding = String.copyValueOf(spaces);
338         String top = format("%sDescription[ t:%s n:%s c:%s %s%n%s   %s] ", padding, type, name, isCollection,
339                 childClass, padding, desc);
340         if (children.isEmpty()) {
341             return top;
342         }
343         StringBuilder sb = new StringBuilder(top);
344         for (Description child : children.values()) {
345             sb.append(System.lineSeparator()).append(child.toString(indent + 2));
346         }
347         return sb.toString();
348     }
