291     /**
292      * Parses the standard options to create a ReportConfiguraton.
293      *
294      * @param args    the arguments to parse
295      * @param helpCmd the help command to run when necessary.
296      * @param noArgs  If true then the commands do not need extra arguments
297      * @return a ReportConfiguration or null if Help was printed.
298      * @throws IOException on error.
299      */
300     public static ReportConfiguration parseCommands(String[] args, Consumer<Options> helpCmd, boolean noArgs) throws IOException {
301         Options opts = buildOptions();
302         CommandLine cl;
303         try {
304             cl = new DefaultParser().parse(opts, args);
305         } catch (ParseException e) {
306             DefaultLog.INSTANCE.error(e.getMessage());
307             DefaultLog.INSTANCE.error("Please use the \"--help\" option to see a list of valid commands and options");
308             System.exit(1);
309             return null; // dummy return (won't be reached) to avoid Eclipse complaint about possible NPE
310             // for "cl"
311         }
312 
313         if (cl.hasOption(LOG_LEVEL)) {
314             try {
315                 DefaultLog.INSTANCE.setLevel(cl.getParsedOptionValue(LOG_LEVEL));
316             } catch (ParseException e) {
317                 logParseException(e, LOG_LEVEL, cl, DefaultLog.INSTANCE.getLevel());
318             }
319         }
320         if (cl.hasOption(HELP)) {
321             helpCmd.accept(opts);
322             return null;
323         }
324 
325         if (!noArgs) {
326             args = cl.getArgs();
327             if (args == null || args.length != 1) {
328                 helpCmd.accept(opts);
329                 return null;
330             }
331         } else {
332             args = new String[]{null};
333         }
334         return createConfiguration(args[0], cl);
335     }

337     private static void logParseException(ParseException e, Option opt, CommandLine cl, Object dflt) {
338         DefaultLog.INSTANCE.warn(format("Invalid %s specified: %s ", opt.getOpt(), cl.getOptionValue(opt)));
339         DefaultLog.INSTANCE.warn(format("%s set to: %s", opt.getOpt(), dflt));
340     }

342     static ReportConfiguration createConfiguration(String baseDirectory, CommandLine cl) throws IOException {
343         final ReportConfiguration configuration = new ReportConfiguration(DefaultLog.INSTANCE);
344 
345         configuration.setDryRun(cl.hasOption(DRY_RUN));
346         if (cl.hasOption(LIST_FAMILIES)) {
347             try {
348                 configuration.listFamilies(cl.getParsedOptionValue(LIST_FAMILIES));
349             } catch (ParseException e) {
350                 logParseException(e, LIST_FAMILIES, cl, Defaults.LIST_FAMILIES);
351             }
352         }
353 
354         if (cl.hasOption(LIST_LICENSES)) {
355             try {
356                 configuration.listLicenses(cl.getParsedOptionValue(LIST_LICENSES));
357             } catch (ParseException e) {
358                 logParseException(e, LIST_LICENSES, cl, Defaults.LIST_LICENSES);
359             }
360         }
361 
362         if (cl.hasOption(ARCHIVE)) {
363             try {
364                 configuration.setArchiveProcessing(cl.getParsedOptionValue(ARCHIVE));
365             } catch (ParseException e) {
366                 logParseException(e, ARCHIVE, cl, Defaults.ARCHIVE_PROCESSING);
367             }
368         }
369 
370         if (cl.hasOption(OUT)) {
371             try {
372                 configuration.setOut((File) cl.getParsedOptionValue(OUT));
373             } catch (ParseException e) {
374                 logParseException(e, OUT, cl, "System.out");
375             }
376         }
377 
378         if (cl.hasOption(SCAN_HIDDEN_DIRECTORIES)) {
379             configuration.setDirectoriesToIgnore(FalseFileFilter.FALSE);
380         }
381 
382         if (ADD.getSelected() != null) {
383             configuration.setAddLicenseHeaders(cl.hasOption(FORCE) ? AddLicenseHeaders.FORCED : AddLicenseHeaders.TRUE);
384             configuration.setCopyrightMessage(cl.getOptionValue(COPYRIGHT));
385         }
386 
387         if (cl.hasOption(EXCLUDE_CLI)) {
388             String[] excludes = cl.getOptionValues(EXCLUDE_CLI);
389             if (excludes != null) {
390                 final FilenameFilter filter = parseExclusions(Arrays.asList(excludes));
391                 configuration.setFilesToIgnore(filter);
392             }
393         } else if (cl.hasOption(EXCLUDE_FILE_CLI)) {
394             String excludeFileName = cl.getOptionValue(EXCLUDE_FILE_CLI);
395             if (excludeFileName != null) {
396                 final FilenameFilter filter = parseExclusions(
397                         FileUtils.readLines(new File(excludeFileName), StandardCharsets.UTF_8));
398                 configuration.setFilesToIgnore(filter);
399             }
400         }
401 
402         if (cl.hasOption(XML)) {
403             configuration.setStyleReport(false);
404         } else {
405             configuration.setStyleReport(true);
406             if (cl.hasOption(STYLESHEET_CLI)) {
407                 String[] style = cl.getOptionValues(STYLESHEET_CLI);
408                 if (style.length != 1) {
409                     DefaultLog.INSTANCE.error("Please specify a single stylesheet");
410                     System.exit(1);
411                 }
412 
413                 URL url = Report.class.getClassLoader().getResource(String.format("org/apache/rat/%s.xsl", style[0]));
414 
415                 IOSupplier<InputStream> ioSupplier = (url == null) ?
416                         () -> Files.newInputStream(Paths.get(style[0])) :
417                         url::openStream;
418                 configuration.setStyleSheet(ioSupplier);
419             }
420         }
421 
422         Defaults.Builder defaultBuilder = Defaults.builder();
423         if (cl.hasOption(NO_DEFAULTS)) {
424             defaultBuilder.noDefault();
425         }
426         if (cl.hasOption(LICENSES)) {
427             for (String fn : cl.getOptionValues(LICENSES)) {
428                 defaultBuilder.add(fn);
429             }
430         }
431         Defaults defaults = defaultBuilder.build(DefaultLog.INSTANCE);
432         configuration.setFrom(defaults);
433         if (StringUtils.isNotBlank(baseDirectory)) {
434             configuration.setReportable(getDirectory(baseDirectory, configuration));
435         }
436         return configuration;
437     }

439     /**
440      * Creates a filename filter from patterns to exclude.
441      *
442      * @param excludes the list of patterns to exclude.
443      * @return the FilenameFilter tht excludes the patterns
444      */
445     static FilenameFilter parseExclusions(List<String> excludes) {
446         final OrFileFilter orFilter = new OrFileFilter();
447         int ignoredLines = 0;
448         for (String exclude : excludes) {
449             try {
450                 // skip comments
451                 if (exclude.startsWith("#") || StringUtils.isEmpty(exclude)) {
452                     ignoredLines++;
453                     continue;
454                 }
455 
456                 String exclusion = exclude.trim();
457                 // interpret given patterns as regular expression, direct file names or
458                 // wildcards to give users more choices to configure exclusions
459                 orFilter.addFileFilter(new RegexFileFilter(exclusion));
460                 orFilter.addFileFilter(new NameFileFilter(exclusion));
461                 orFilter.addFileFilter(WildcardFileFilter.builder().setWildcards(exclusion).get());
462             } catch (PatternSyntaxException e) {
463                 DefaultLog.INSTANCE.error("Will skip given exclusion '" + exclude + "' due to " + e);
464             }
465         }
466         DefaultLog.INSTANCE.error("Ignored " + ignoredLines + " lines in your exclusion files as comments or empty lines.");
467         return new NotFileFilter(orFilter);
468     }

502    static String header(String txt) {
503         return String.format("%n====== %s ======%n", WordUtils.capitalizeFully(txt));

505     static void printUsage(PrintWriter writer, Options opts) {
506 
507         HelpFormatter helpFormatter = new HelpFormatter();
508         helpFormatter.setWidth(130);
509         helpFormatter.setOptionComparator(new OptionComparator());
510         String syntax = "java -jar apache-rat/target/apache-rat-CURRENT-VERSION.jar [options] [DIR|TARBALL]";
511         helpFormatter.printHelp(writer, helpFormatter.getWidth(), syntax, header("Available options"), opts,
512                 helpFormatter.getLeftPadding(), helpFormatter.getDescPadding(),
513                 header("Argument Types"), false);
514 
515         String argumentPadding = createPadding(helpFormatter.getLeftPadding() + 5);
516         for (Map.Entry<String, Supplier<String>> argInfo : ARGUMENT_TYPES.entrySet()) {
517             writer.format("\n<%s>\n", argInfo.getKey());
518             helpFormatter.printWrapped(writer, helpFormatter.getWidth(), helpFormatter.getLeftPadding() + 10,
519                     argumentPadding + argInfo.getValue().get());
520         }
521         writer.println(header("Notes"));
522 
523         int idx = 1;
524         for (String note : NOTES) {
525             writer.format("%d. %s%n", idx++, note);
526         }
527         writer.flush();
528     }

534     /**
535      * Creates an IReportable object from the directory name and ReportConfiguration
536      * object.
537      *
538      * @param baseDirectory the directory that contains the files to report on.
539      * @param config        the ReportConfiguration.
540      * @return the IReportable instance containing the files.
541      */
542     private static IReportable getDirectory(String baseDirectory, ReportConfiguration config) {
543         File base = new File(baseDirectory);
544 
545         if (!base.exists()) {
546             config.getLog().log(Level.ERROR, "Directory '" + baseDirectory + "' does not exist");
547             return null;
548         }
549 
550         Document doc = new FileDocument(base);
551         if (base.isDirectory()) {
552             return new DirectoryWalker(config, doc);
553         }
554 
555         return new ArchiveWalker(config, doc);
556     }

558     /**
559      * This class implements the {@code Comparator} interface for comparing Options.
560      */
561     public static class OptionComparator implements Comparator<Option>, Serializable {
562         /** The serial version UID.  */
563         private static final long serialVersionUID = 5305467873966684014L;
564 
565         private String getKey(Option opt) {
566             String key = opt.getOpt();
567             key = key == null ? opt.getLongOpt() : key;
568             return key;
569         }
570 
571         /**
572          * Compares its two arguments for order. Returns a negative integer, zero, or a
573          * positive integer as the first argument is less than, equal to, or greater
574          * than the second.
575          *
576          * @param opt1 The first Option to be compared.
577          * @param opt2 The second Option to be compared.
578          * @return a negative integer, zero, or a positive integer as the first argument
579          * is less than, equal to, or greater than the second.
580          */
581         @Override
582         public int compare(final Option opt1, final Option opt2) {
583             return getKey(opt1).compareToIgnoreCase(getKey(opt2));
584         }
585     }
