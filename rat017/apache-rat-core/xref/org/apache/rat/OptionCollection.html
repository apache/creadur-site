125     /**
126      * Parses the standard options to create a ReportConfiguration.
127      *

130      * @return a ReportConfiguration or null if Help was printed.

137     /**
138      * Parses the standard options to create a ReportConfiguration.
139      *
140      * @param args the arguments to parse
141      * @param helpCmd the help command to run when necessary.
142      * @param noArgs If true then the commands do not need extra arguments
143      * @return a ReportConfiguration or null if Help was printed.
144      * @throws IOException on error.
145      */
146     public static ReportConfiguration parseCommands(final String[] args, final Consumer<Options> helpCmd, final boolean noArgs) throws IOException {
147         Options opts = buildOptions();
148         CommandLine commandLine;
149         try {
150             commandLine = DefaultParser.builder().setDeprecatedHandler(DeprecationReporter.getLogReporter())
151                     .setAllowPartialMatching(true).build().parse(opts, args);
152         } catch (ParseException e) {
153             DefaultLog.getInstance().error(e.getMessage());
154             DefaultLog.getInstance().error("Please use the \"--help\" option to see a list of valid commands and options", e);
155             System.exit(1);
156             return null; // dummy return (won't be reached) to avoid Eclipse complaint about possible NPE
157             // for "commandLine"
158         }
159 
160         Arg.processLogLevel(commandLine);
161 
162         if (commandLine.hasOption(HELP)) {
163             helpCmd.accept(opts);
164             return null;
165         }
166 
167         if (commandLine.hasOption(HELP_LICENSES)) {
168             new Licenses(createConfiguration(null, commandLine), new PrintWriter(System.out)).printHelp();
169             return null;
170         }
171 
172         if (commandLine.hasOption(HELP_LICENSES)) {
173             new Licenses(createConfiguration(null, commandLine), new PrintWriter(System.out)).printHelp();
174             return null;
175         }
176 
177         if (commandLine.hasOption(Arg.HELP_LICENSES.option())) {
178             new Licenses(createConfiguration(null, commandLine), new PrintWriter(System.out)).printHelp();
179             return null;
180         }
181 
182         // DIR or end of command line can provide args.
183         String[] clArgs = {null};
184         List<String> lst = new ArrayList<>();
185         String dirValue = commandLine.getOptionValue(Arg.DIR.getSelected());
186         if (dirValue != null) {
187             lst.add(dirValue);
188         }
189         lst.addAll(Arrays.asList(commandLine.getArgs()));
190         if (!noArgs && lst.size() != 1) {
191             helpCmd.accept(opts);
192             return null;
193         }
194         if (!lst.isEmpty()) {
195             clArgs[0] = lst.get(0);
196         }
197         return createConfiguration(clArgs[0], commandLine);
198     }

200     /**
201      * Create the report configuration.
202      * Note: this method is package private for testing. You probably want one of the {@code ParseCommands} methods.
203      * @param baseDirectory the base directory where files will be found.
204      * @param cl the parsed command line.
205      * @return a ReportConfiguration
206      * @see #parseCommands(String[], Consumer)
207      * @see #parseCommands(String[], Consumer, boolean)
208      */
209     static ReportConfiguration createConfiguration(final String baseDirectory, final CommandLine cl) {
210         final ReportConfiguration configuration = new ReportConfiguration();
211         new ArgumentContext(configuration, cl).processArgs();
212         if (StringUtils.isNotBlank(baseDirectory)) {
213             configuration.setReportable(getReportable(new File(baseDirectory), configuration));
214         }
215         return configuration;
216     }

227     /**
228      * Creates an IReportable object from the directory name and ReportConfiguration
229      * object.
230      *
231      * @param base the directory that contains the files to report on.
232      * @param config the ReportConfiguration.
233      * @return the IReportable instance containing the files.
234      */
235     static IReportable getReportable(final File base, final ReportConfiguration config) {
236         File absBase = base.getAbsoluteFile();
237         DocumentName documentName = new DocumentName(absBase);
238         if (!absBase.exists()) {
239             DefaultLog.getInstance().error("Directory '" + documentName + "' does not exist");
240             return null;
241         }
242         DocumentNameMatcher documentNameMatcher = config.getNameMatcher(documentName);
243 
244         Document doc = new FileDocument(documentName, absBase, documentNameMatcher);
245         if (!documentNameMatcher.matches(doc.getName())) {
246             DefaultLog.getInstance().error("Directory '" + documentName + "' is in excluded list.");
247             return null;
248         }
249 
250         if (absBase.isDirectory()) {
251             return new DirectoryWalker(doc);
252         }
253 
254         return new ArchiveWalker(doc);
255     }

257     /**
258      * This class implements the {@code Comparator} interface for comparing Options.
259      */
260     private static class OptionComparator implements Comparator<Option>, Serializable {
261         /** The serial version UID.  */
262         private static final long serialVersionUID = 5305467873966684014L;
263 
264         private String getKey(final Option opt) {
265             String key = opt.getOpt();
266             key = key == null ? opt.getLongOpt() : key;
267             return key;
268         }
269 
270         /**
271          * Compares its two arguments for order. Returns a negative integer, zero, or a
272          * positive integer as the first argument is less than, equal to, or greater
273          * than the second.
274          *
275          * @param opt1 The first Option to be compared.
276          * @param opt2 The second Option to be compared.
277          * @return a negative integer, zero, or a positive integer as the first argument
278          * is less than, equal to, or greater than the second.
279          */
280         @Override
281         public int compare(final Option opt1, final Option opt2) {
282             return getKey(opt1).compareToIgnoreCase(getKey(opt2));
283         }
284     }
