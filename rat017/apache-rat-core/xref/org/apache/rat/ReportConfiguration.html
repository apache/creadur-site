392     /**
393      * Sets the supplier for the output stream. The supplier may be called multiple
394      * times to provide the stream. Suppliers should prepare streams that are
395      * appended to and that can be closed. If an {@code OutputStream} should not be
396      * closed consider wrapping it in a {@code NoCloseOutputStream}
397      * @param out The OutputStream supplier that provides the output stream to write
398      * the report to. A null value will use System.out.
399      * @see NoCloseOutputStream
400      */
401     public void setOut(final IOSupplier<OutputStream> out) {
402         this.out = out;
403     }
404 
405     /**
406      * Sets the OutputStream supplier to use the specified file. The file may be
407      * opened and closed several times. File is deleted first and then may be
408      * repeatedly opened in append mode.
409      * @see #setOut(IOSupplier)
410      * @param file The file to create the supplier with.
411      */
412     public void setOut(final File file) {
413         Objects.requireNonNull(file, "output file should not be null");
414         if (file.exists()) {
415             try {
416                 Files.delete(file.toPath());
417             } catch (IOException e) {
418                 DefaultLog.getInstance().warn("Unable to delete file: " + file);
419             }
420         }
421         if (!file.getParentFile().mkdirs()) {
422             DefaultLog.getInstance().warn("Unable to create directory: " + file.getParentFile());
423         }
424         setOut(() -> new FileOutputStream(file, true));
425     }

672             break;
673         case FORCED:

677         case TRUE:
678             addingLicenses = true;
679             break;
680         }

711     /**
712      * A wrapper around an output stream that does not close the output stream.
713      */
714     public static class NoCloseOutputStream extends OutputStream {
715         /** the output stream this stream wraps */
716         private final OutputStream delegate;
717 
718         /**
719          * Constructor.
720          * @param delegate the output stream to wrap.
721          */
722         public NoCloseOutputStream(final OutputStream delegate) {
723             this.delegate = delegate;
724         }
725 
726         @Override
727         public void write(final int arg0) throws IOException {
728             delegate.write(arg0);
729         }
730 
731         /**
732          * Does not actually close the delegate.  But does perform a flush.
733          * @throws IOException on Error.
734          */
735         @Override
736         public void close() throws IOException {
737             this.delegate.flush();
738         }
739 
740         @Override
741         public boolean equals(final Object obj) {
742             return delegate.equals(obj);
743         }
744 
745         @Override
746         public void flush() throws IOException {
747             delegate.flush();
748         }
