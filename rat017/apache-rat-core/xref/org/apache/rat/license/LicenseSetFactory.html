33  /**
34   * Class to take a set of ILicenses and collection of approved license
35   * categories and extract Subsets.
36   */
37  public class LicenseSetFactory {
38  
39      /**
40       * Search a SortedSet of ILicenseFamily instances looking for a matching instance.
41       * @param target The instance to search for.
42       * @param licenseFamilies the license families to search
43       * @return the matching instance of the target given.
44       */
45      public static ILicenseFamily familySearch(final String target, final SortedSet<ILicenseFamily> licenseFamilies) {
46          ILicenseFamily family = ILicenseFamily.builder().setLicenseFamilyCategory(target).setLicenseFamilyName("Searching family")
47                  .build();
48          return familySearch(family, licenseFamilies);
49      }
50  
51      /**
52       * Search a SortedSet of ILicenseFamily instances looking for a matching instance.
53       * @param target The instance to search for.
54       * @param licenseFamilies the license families to search
55       * @return the matching instance of the target given.
56       */
57      public static ILicenseFamily familySearch(final ILicenseFamily target, final SortedSet<ILicenseFamily> licenseFamilies) {
58          SortedSet<ILicenseFamily> part = licenseFamilies.tailSet(target);
59          return (!part.isEmpty() && part.first().compareTo(target) == 0) ? part.first() : null;
60      }

62      /**
63       * An enum that defines the types of Licenses to extract.
64       */
65      public enum LicenseFilter {
66          /** All defined licenses are returned */
67          ALL,
68          /** Only approved licenses are returned */
69          APPROVED,
70          /** No licenses are returned */
71          NONE;
72      }
73  
74      /** The set of defined families. */
75      private final ReportingSet<ILicenseFamily> families;
76      /** The set of defined licenses */
77      private final ReportingSet<ILicense> licenses;
78  
79      /** The set of approved license family categories. If the category is not listed, the family is not approved. */
80      private final SortedSet<String> approvedLicenseCategories;
81      /**
82       * The set of license categories that are to be removed from consideration. These are categories that were
83       * added but should now be removed.
84       */
85      private final SortedSet<String> removedLicenseCategories;
86      /**
87       * The set of approved license ids.  This set contains the set of licenses that are explicitly approved even if
88       * the family is not.
89       */
90      private final SortedSet<String> approvedLicenseIds;
91      /**
92       * The set of license ids that are to be removed from consideration. This set contains licenses that are to be
93       * removed even if the family is approved or if an earlier license approval was granted.
94       */
95      private final SortedSet<String> removedLicenseIds;
96  
97      /**
98       * Constructs a factory with the specified set of Licenses and the approved
99       * license collection.
100      */
101     public LicenseSetFactory() {
102         families = new ReportingSet<>(new TreeSet<ILicenseFamily>())
103                 .setMsgFormat(s -> String.format("Duplicate LicenseFamily category: %s", s.getFamilyCategory()));
104         licenses = new ReportingSet<>(new TreeSet<ILicense>())
105                 .setMsgFormat(s -> String.format("Duplicate License %s (%s) of type %s", s.getName(), s.getId(), s.getLicenseFamily().getFamilyCategory()));
106 
107         approvedLicenseCategories = new TreeSet<>();
108         removedLicenseCategories = new TreeSet<>();
109         approvedLicenseIds = new TreeSet<>();
110         removedLicenseIds = new TreeSet<>();
111     }
112 
113     /**
114      * Constructs a factory with the specified set of Licenses and the approved
115      * license collection.
116      * @param licenses the set of defined licenses. Families will be extracted from the licenses.
117      */
118     public LicenseSetFactory(final SortedSet<ILicense> licenses) {
119         this();
120         this.licenses.addAll(licenses);
121         licenses.forEach(l -> families.addIfNotPresent(l.getLicenseFamily()));
122     }
123 
124     public void add(final LicenseSetFactory other) {
125         this.families.addAll(other.families);
126         this.licenses.addAll(other.licenses);
127         this.approvedLicenseCategories.addAll(other.approvedLicenseCategories);
128         this.removedLicenseCategories.addAll(other.removedLicenseCategories);
129         this.approvedLicenseIds.addAll(other.approvedLicenseIds);
130         this.removedLicenseIds.addAll(other.removedLicenseIds);
131     }

166     /**
167      * Create a sorted set of licenses families from the collection.
168      * @param licenses the collection of all licenses.
169      * @return a SortedSet of license families from the collection.
170      */
171     private static SortedSet<ILicenseFamily> extractFamily(final Collection<ILicense> licenses) {
172         SortedSet<ILicenseFamily> result = new TreeSet<>();
173         licenses.stream().map(ILicense::getLicenseFamily).forEach(result::add);
174         return result;
175     }

177     /**
178      * Adds a license to the list of licenses. Does not add the license to the list
179      * of approved licenses.
180      * @param license The license to add to the list of licenses.
181      */
182     public void addLicense(final ILicense license) {
183         if (license != null) {
184             this.licenses.add(license);
185             this.families.addIfNotPresent(license.getLicenseFamily());
186         }
187     }

189     /**
190      * Adds a license to the list of licenses. Does not add the license to the list
191      * of approved licenses.
192      * @param builder The license builder to build and add to the list of licenses.
193      * @return The ILicense implementation that was added.
194      */
195     public ILicense addLicense(final ILicense.Builder builder) {
196         if (builder != null) {
197             ILicense license = builder.setLicenseFamilies(families).build();
198             this.licenses.add(license);
199             return license;
200         }
201         return null;
202     }

269     /**
270      * Test for approved family category.
271      * @param family the license family to test, must be in category format.
272      * @return return {@code true} if the category is approved.
273      */
274     private boolean isApprovedCategory(final ILicenseFamily family) {
275         return approvedLicenseCategories.contains(family.getFamilyCategory()) && !removedLicenseCategories.contains(family.getFamilyCategory());
276     }

278     /**
279      * Gets the License objects based on the filter.
280      * @param filter the types of LicenseFamily objects to return.
281      * @return a SortedSet of ILicense objects.
282      */
283     public SortedSet<ILicense> getLicenses(final LicenseFilter filter) {
284         Predicate<ILicense> approved =  l -> (isApprovedCategory(l.getLicenseFamily()) ||
285                 approvedLicenseIds.contains(l.getId())) && !removedLicenseIds.contains(l.getId());
286 
287         switch (filter) {
288         case ALL:
289             return Collections.unmodifiableSortedSet(licenses);
290         case APPROVED:
291             SortedSet<ILicense> result = new TreeSet<>();
292             licenses.stream().filter(approved).forEach(result::add);
293             return result;
294         case NONE:
295         default:
296             return Collections.emptySortedSet();
297         }
298     }
299 
300     /**
301      * Gets the LicenseFamily objects based on the filter.
302      * @param filter the types of LicenseFamily objects to return.
303      * @return a SortedSet of ILicenseFamily objects.
304      */
305     public SortedSet<ILicenseFamily> getLicenseFamilies(final LicenseFilter filter) {
306         SortedSet<ILicenseFamily> result;
307         switch (filter) {
308         case ALL:
309             result = extractFamily(licenses);
310             result.addAll(families);
311             return result;
312         case APPROVED:
313             result = new TreeSet<>();
314             licenses.stream().map(ILicense::getLicenseFamily).filter(this::isApprovedCategory).forEach(result::add);
315             return result;
316         case NONE:
317         default:
318             return Collections.emptySortedSet();
319         }
320     }

322     /**
323      * Gets the License ids based on the filter.
324      *
325      * @param filter the types of License IDs to return.
326      * @return The list of all licenses in the category regardless of whether or not it is used by an ILicense implementation.
327      */
328     public SortedSet<String> getLicenseCategories(final LicenseFilter filter) {
329         Predicate<ILicense> approved = l -> (isApprovedCategory(l.getLicenseFamily()) ||
330                 approvedLicenseIds.contains(l.getId())) && !removedLicenseIds.contains(l.getId());
331         SortedSet<String> result = new TreeSet<>();
332         switch (filter) {
333             case ALL:
334                 licenses.forEach(l -> result.add(l.getLicenseFamily().getFamilyCategory()));
335                 families.forEach(f -> result.add(f.getFamilyCategory()));
336                 result.addAll(approvedLicenseCategories);
337                 result.addAll(removedLicenseCategories);
338                 return result;
339             case APPROVED:
340                 approvedLicenseCategories.stream().filter(s -> !removedLicenseCategories.contains(s)).forEach(result::add);
341                 licenses.stream().filter(approved).forEach(l -> result.add(l.getLicenseFamily().getFamilyCategory()));
342                 families.stream().filter(this::isApprovedCategory).forEach(f -> result.add(f.getFamilyCategory()));
343                 return result;
344             case NONE:
345             default:
346                 return Collections.emptySortedSet();
347         }
348     }
349 
350     /**
351      * Gets the License ids based on the filter.
352      *
353      * @param filter the types of License IDs to return.
354      * @return The list of all licenses in the category regardless of whether or not it is used by an ILicense implementation.
355      */
356     public SortedSet<String> getLicenseIds(final LicenseFilter filter) {
357         Predicate<ILicense> approved =  l -> (isApprovedCategory(l.getLicenseFamily()) ||
358                 approvedLicenseIds.contains(l.getId())) && !removedLicenseIds.contains(l.getId());
359         SortedSet<String> result = new TreeSet<>();
360         switch (filter) {
361             case ALL:
362                 licenses.forEach(l -> result.add(l.getId()));
363                 result.addAll(approvedLicenseCategories);
364                 result.addAll(removedLicenseCategories);
365                 result.addAll(approvedLicenseIds);
366                 result.addAll(removedLicenseIds);
367                 return result;
368             case APPROVED:
369                 licenses.stream().filter(approved).forEach(l -> result.add(l.getId()));
370                 families.stream().filter(this::isApprovedCategory).forEach(f -> result.add(f.getFamilyCategory()));
371                 approvedLicenseIds.stream().filter(s -> !removedLicenseIds.contains(s)).forEach(result::add);
372                 return result;
373             case NONE:
374             default:
375                 return Collections.emptySortedSet();
376         }
377     }

379     /**
380      * Search a SortedSet of licenses for the matching license id.
381      *
382      * @param licenseId the id to search for.
383      * @param licenses the SortedSet of licenses to search.
384      * @return the matching license or {@code null} if not found.
385      */
386     public static Optional<ILicense> search(final String familyId, final String licenseId, final SortedSet<ILicense> licenses) {
387         ILicenseFamily searchFamily = ILicenseFamily.builder().setLicenseFamilyCategory(familyId)
388                 .setLicenseFamilyName("searching proxy").build();
389         ILicense target = new ILicense() {
390 
391             @Override
392             public String getId() {
393                 return licenseId;
394             }
395 
396             @Override
397             public void reset() {
398                 // do nothing
399             }
400 
401             @Override
402             public boolean matches(final IHeaders headers) {
403                 return false;
404             }
405 
406             @Override
407             public boolean equals(final Object o) {
408                 return ILicense.equals(this, o);
409             }
410 
411             @Override
412             public int hashCode() {
413                 return ILicense.hash(this);
414             }
415 
416             @Override
417             public ILicenseFamily getLicenseFamily() {
418                 return searchFamily;
419             }
420 
421             @Override
422             public String getNote() {
423                 return null;
424             }
425 
426             @Override
427             public String getName() {
428                 return searchFamily.getFamilyName();
429             }
430 
431             @Override
432             public IHeaderMatcher getMatcher() {
433                 return null;
434             }
435 
436         };
437         return search(target, licenses);
438     }

440     /**
441      * Search a SortedSet of licenses for the matching license.
442      * License must mach both family code, and license id.
443      *
444      * @param target the license to search for. Must not be null.
445      * @param licenses the SortedSet of licenses to search.
446      * @return the matching license or {@code null} if not found.
447      */
448     public static Optional<ILicense> search(final ILicense target, final SortedSet<ILicense> licenses) {
449         SortedSet<ILicense> part = licenses.tailSet(target);
450         return Optional.ofNullable((!part.isEmpty() && part.first().compareTo(target) == 0) ? part.first() : null);
451     }
