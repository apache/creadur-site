52  /**
53   * A basic Ant task that generates a report on all files specified by the nested
54   * resource collection(s).
55   *
56   * <p>
57   * IHeaderMatcher(s) can be specified as nested elements as well.
58   * </p>
59   *
60   * <p>
61   * The attribute <code>format</code> defines the output format and can take the
62   * values
63   * <ul>
64   * <li>xml - Rat's native XML output.</li>
65   * <li>styled - transforms the XML output using the given stylesheet. The
66   * stylesheet attribute must be set as well if this attribute is used.</li>
67   * <li>plain - plain text using Rat's built-in stylesheet. This is the
68   * default.</li>
69   * </ul>
70   */

313     /**
314      * Reads values for the Arg.
315      *
316      * @param arg The Arg to get the values for.
317      * @return The list of values or an empty list.
318      */
319     protected List<String> getValues(final Arg arg) {
320         List<String> result = new ArrayList<>();
321         for (Option option : arg.group().getOptions()) {
322             if (option.getLongOpt() != null) {
323                 List<String> args = getArg(option.getLongOpt());
324                 if (args != null) {
325                     result.addAll(args);
326                 }
327             }
328         }
329         return result;
330     }
331 
332     /**
333      * Removes the values for the arg.
334      * @param arg the arg to remove the values for.
335      */
336     protected void removeKey(final Arg arg) {
337         for (Option option : arg.group().getOptions()) {
338             if (option.getLongOpt() != null) {
339                 removeArg(option.getLongOpt());
340             }
341         }
342     }

345     /**
346      * Creates the ReportConfiguration from the ant options.
347      * @return the ReportConfiguration.
348      */
349     public ReportConfiguration getConfiguration() {
350         try {
351             boolean helpLicenses = !getValues(Arg.HELP_LICENSES).isEmpty();
352             removeKey(Arg.HELP_LICENSES);
353 
354             final ReportConfiguration configuration = OptionCollection.parseCommands(args().toArray(new String[0]),
355                     o -> DefaultLog.getInstance().warn("Help option not supported"),
356                     true);
357             if (getValues(Arg.OUTPUT_FILE).isEmpty()) {
358                 configuration.setOut(() -> new LogOutputStream(this, Project.MSG_INFO));
359             }
360             configuration.setReportable(new ResourceCollectionContainer(nestedResources));
361             configuration.addApprovedLicenseCategories(deprecatedConfig.approvedLicenseCategories);
362             configuration.removeApprovedLicenseCategories(deprecatedConfig.removedLicenseCategories);
363             if (deprecatedConfig.inputFileFilter != null) {
364                 if (configuration.getFilesToIgnore() != null) {
365                     configuration.setFilesToIgnore(new OrFileFilter(configuration.getFilesToIgnore(), deprecatedConfig.inputFileFilter));
366                 } else {
367                     configuration.setFilesToIgnore(deprecatedConfig.inputFileFilter);
368                 }
369             }
370             families.stream().map(Family::build).forEach(configuration::addFamily);
371             licenses.stream().map(License::asBuilder)
372                     .forEach(l -> configuration.addApprovedLicenseCategory(configuration.addLicense(l).getLicenseFamily()));
373             if (helpLicenses) {
374                 new org.apache.rat.help.Licenses(configuration, new PrintWriter(DefaultLog.getInstance().asWriter())).printHelp();
375             }
376             return configuration;
377         } catch (IOException | ImplementationException e) {
378             throw new BuildException(e.getMessage(), e);
379         }
380     }

382     /**
383      * Generates the report.
384      */
385     @Override
386     public void execute() {
387         try {
388             Reporter r = new Reporter(validate(getConfiguration()));
389             r.output(StyleSheets.PLAIN.getStyleSheet(), () -> new ReportConfiguration.NoCloseOutputStream(System.out));
390             r.output();
391         } catch (BuildException e) {
392             throw e;
393         } catch (Exception ioex) {
394             throw new BuildException(ioex);
395         }
396     }
397 
398     /**
399      * validates the task's configuration.
400      */
401     protected ReportConfiguration validate(final ReportConfiguration cfg) {
402         try {
403             cfg.validate(s -> log(s, Project.MSG_WARN));
404         } catch (ConfigurationException e) {
405             throw new BuildException(e.getMessage(), e.getCause());
406         }
407         if (nestedResources == null) {
408             throw new BuildException("You must specify at least one file to create the report for.");
409         }
410         return cfg;
411     }

468     /**
469      * A facade for the Logger provided by Ant.
470      */
471     private class Logger implements Log {
472 
473         private void write(final int level, final String msg) {
474             try (PrintWriter pw = new PrintWriter(new LogOutputStream(Report.this, level))) {
475                pw.write(msg);
476             }
477         }
478 
479         @Override
480         public void log(final Level level, final String msg) {
481             switch (level) {
482             case DEBUG:
483                 write(Project.MSG_DEBUG, msg);
484                 break;
485             case INFO:
486                 write(Project.MSG_INFO, msg);
487                 break;
488             case WARN:
489                 write(Project.MSG_WARN, msg);
490                 break;
491             case ERROR:
492                 write(Project.MSG_ERR, msg);
493                 break;
494             case OFF:
495             default:
496                 break;
497             }
498         }
