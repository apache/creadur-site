65      /**
66       * Creates a new instance of {@link AbstractRatMojo}.
67       *
68       * @param pDir The directory, where to look for a pom.xml file.
69       * @param pGoal The goal, which the Mojo must implement.
70       * @param pCreateCopy if {@code true} copy the directory contents and return the
71       * copy location.
72       * @return The configured Mojo.
73       * @throws Exception An error occurred while creating the Mojo.
74       */
75      private AbstractRatMojo newRatMojo(String pDir, String pGoal, boolean pCreateCopy) throws Exception {
76          final File baseDir = new File(getBasedir());
77          final File testBaseDir = getSourceDirectory(getBasedir(), pDir, pCreateCopy, baseDir);
78          final File testPom = new File(testBaseDir, "pom.xml");
79          final File buildDirectory = new File(new File(baseDir, "target/test"), pDir);
80          AbstractRatMojo mojo = (AbstractRatMojo) lookupConfiguredMojo(testPom, pGoal);
81          assertNotNull(mojo);
82  
83          assertNotNull("The mojo is missing its MavenProject, which will result in an NPE during rat runs.",
84                  mojo.getProject());
85  
86          if (mojo instanceof RatReportMojo) {
87              setVariableValueToObject(mojo, "localRepository", newArtifactRepository(getContainer()));
88              setVariableValueToObject(mojo, "factory", newArtifactFactory());
89              setVariableValueToObject(mojo, "siteRenderer", newSiteRenderer(getContainer()));
90          } else if (mojo instanceof RatCheckMojo) {
91              final File ratTxtFile = new File(buildDirectory, "rat.txt");
92              FileUtils.write(ratTxtFile, "", UTF_8); // Ensure the output file exists and is empty (rerunning the test will append)
93              mojo.setOutputFile(ratTxtFile.getAbsolutePath());
94          }
95          return mojo;
96      }
97  
98      private String getDir(RatCheckMojo mojo) {
99          return mojo.getProject().getBasedir().getAbsolutePath().replace("\\","/") + "/";
100     }

178     /**
179      * Tests defining licenses in configuration
180      */
181     public void testIt5() throws Exception {
182         final RatCheckMojo mojo = (RatCheckMojo) newRatMojo("it5", "check", true);
183         final File ratTxtFile = mojo.getRatTxtFile();
184         final String[] expected = { "Notes: 0", "Binaries: 0", "Archives: 0", "Standards: 0$", "Apache Licensed: 0$",
185                 "Generated Documents: 0", "^0 Unknown Licenses" };
186 
187         ReportConfiguration config = mojo.getConfiguration();
188         assertFalse("Should not be adding licenses", config.isAddingLicenses());
189         assertFalse("Should not be forcing licenses", config.isAddingLicensesForced());
190 
191         ReportConfigurationTest.validateDefaultApprovedLicenses(config, 1);
192         assertTrue(config.getLicenseCategories(LicenseFilter.APPROVED).contains(ILicenseFamily.makeCategory("YAL")));
193         ReportConfigurationTest.validateDefaultLicenseFamilies(config, "YAL");
194         assertNotNull(LicenseSetFactory.familySearch("YAL", config.getLicenseFamilies(LicenseFilter.ALL)));
195         ReportConfigurationTest.validateDefaultLicenses(config, "MyLicense", "CpyrT", "RegxT", "SpdxT", "TextT", 
196                 "Not", "All", "Any");
197         assertTrue(LicenseSetFactory.search("YAL", "MyLicense", config.getLicenses(LicenseFilter.ALL)).isPresent());
198         assertNotNull("Should have filesToIgnore", config.getFilesToIgnore());
199         assertThat(config.getFilesToIgnore()).isExactlyInstanceOf(FalseFileFilter.class);
200         assertNotNull("Should have directoriesToIgnore", config.getDirectoriesToIgnore());
201         assertThat(config.getDirectoriesToIgnore()).isExactlyInstanceOf(NameBasedHiddenFileFilter.class);
202         mojo.execute();
203 
204         ensureRatReportIsCorrect(ratTxtFile, expected, TextUtils.EMPTY);
205     }

212     public void testRAT_343() throws Exception {
213         final RatCheckMojo mojo = newRatCheckMojo("RAT-343");
214         final File ratTxtFile = mojo.getRatTxtFile();
215         // POM reports AL, BSD and CC BYas BSD because it contains the BSD and CC BY strings
216         final String[] expected = { 
217                 ReporterTestUtils.documentOut(false, Document.Type.STANDARD, getDir(mojo) + "pom.xml") +
218                 ReporterTestUtils.APACHE_LICENSE +
219                 ReporterTestUtils.licenseOut("BSD", "BSD") +
220                 ReporterTestUtils.licenseOut("CC BY", "Creative Commons Attribution (Unapproved)"),
221                 "Notes: 0", "Binaries: 0", "Archives: 0",
222                 "Standards: 1$", "Apache Licensed: 1$", "Generated Documents: 0", "^0 Unknown Licenses" };
223 
224         ReportConfiguration config = mojo.getConfiguration();
225         // validate configuration
226         assertThat(config.isAddingLicenses()).isFalse();
227         assertThat(config.isAddingLicensesForced()).isFalse();
228         assertThat(config.getCopyrightMessage()).isNull();
229         assertThat(config.getStyleSheet()).withFailMessage("Stylesheet should not be null").isNotNull();
230         assertThat(config.getDirectoriesToIgnore()).withFailMessage("directoriesToIgnore filter should not be null").isNotNull();
231         assertThat(config.getDirectoriesToIgnore()).isExactlyInstanceOf(NameBasedHiddenFileFilter.class);
232         assertThat(config.getFilesToIgnore()).withFailMessage("filesToIgnore filter should not be null").isNotNull();
233         assertThat(config.getFilesToIgnore()).isExactlyInstanceOf(FalseFileFilter.class);
234         
235         ReportConfigurationTest.validateDefaultApprovedLicenses(config, 1);
236         ReportConfigurationTest.validateDefaultLicenseFamilies(config, "BSD", "CC BY");
237         ReportConfigurationTest.validateDefaultLicenses(config, "BSD", "CC BY");
238 
239         mojo.execute();
240         ensureRatReportIsCorrect(ratTxtFile, expected, TextUtils.EMPTY);
241     }

278     /**
279      * Tests verifying gitignore parsing under a special edge case condition
280      * The problem occurs when '/foo.md' is to be ignored and a file with that name exists
281      * in a directory which is the project base directory twice concatenated.
282      * So for this test we must create such a file which is specific for the current
283      * working directory.
284      */
285     public void testRAT362GitIgnore() throws Exception {
286         final RatCheckMojo mojo = newRatCheckMojo("RAT-362-GitIgnore");
287         final File ratTxtFile = mojo.getRatTxtFile();
288         final String dirStr = getDir(mojo);
289 
290         if (dirStr.contains(":")) {
291             // The problem this is testing for cannot happen if there is
292             // a Windows drive letter in the name of the directory.
293             // Any duplication of a ':' will make it all fail always.
294             // So there is no point in continuing this test.
295             return;
296         }
297         File dir = new File(dirStr);
298 
299         // Make the target directory for the test file
300         assertTrue(dir.isDirectory());
301 
302         // Create the test file with a content on which it must fail
303         File barFile = new File(dir, "bar.md");
304         final String[] expected = {
305                 "Notes: 0",
306                 "Binaries: 0",
307                 "Archives: 0",
308                 "Standards: 3$",
309                 "Apache Licensed: 2$",
310                 "Generated Documents: 0",
311                 "^1 Unknown Licenses",
312                 ReporterTestUtils.documentOut(false, Document.Type.STANDARD, barFile.getCanonicalPath()) +
313                         ReporterTestUtils.UNKNOWN_LICENSE
314         };
315         try {
316             mojo.execute();
317             fail("Expected RatCheckException: This check should have failed on the invalid test file");
318         } catch (RatCheckException e) {
319             final String msg = e.getMessage();
320             assertTrue("report filename was not contained in '" + msg + "'", msg.contains(ratTxtFile.getName()));
321             assertFalse("no null allowed in '" + msg + "'", (msg.toUpperCase().contains("NULL")));
322             ensureRatReportIsCorrect(ratTxtFile, expected, TextUtils.EMPTY);
323         }
