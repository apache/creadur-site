48      /**
49       * Creates a new instance of {@link RatCheckMojo}.
50       *
51       * @param pDir The directory, where to look for a pom.xml file.
52       * @return The configured Mojo.
53       * @throws Exception An error occurred while creating the Mojo.

56          return (RatCheckMojo) newRatMojo(pDir, "check", false);
57      }

59      /**
60       * Creates a new instance of {@link AbstractRatMojo}.
61       *
62       * @param pDir The directory, where to look for a pom.xml file.
63       * @param pGoal The goal, which the Mojo must implement.
64       * @param pCreateCopy if {@code true} copy the directory contents and return the
65       * copy location.
66       * @return The configured Mojo.
67       * @throws Exception An error occurred while creating the Mojo.
68       */
69      private AbstractRatMojo newRatMojo(String pDir, String pGoal, boolean pCreateCopy) throws Exception {
70          final File baseDir = new File(getBasedir());
71          final File testBaseDir = getSourceDirectory(getBasedir(), pDir, pCreateCopy, baseDir);
72          final File testPom = new File(testBaseDir, "pom.xml");
73          final File buildDirectory = new File(new File(baseDir, "target/test"), pDir);
74          AbstractRatMojo mojo = (AbstractRatMojo) lookupConfiguredMojo(testPom, pGoal);
75          assertNotNull(mojo);
76  
77          assertNotNull("The mojo is missing its MavenProject, which will result in an NPE during rat runs.",
78                  mojo.getProject());
79  
80          if (mojo instanceof RatReportMojo) {
81              setVariableValueToObject(mojo, "localRepository", newArtifactRepository(getContainer()));
82              setVariableValueToObject(mojo, "factory", newArtifactFactory());
83              setVariableValueToObject(mojo, "siteRenderer", newSiteRenderer(getContainer()));
84          } else if (mojo instanceof RatCheckMojo) {
85              final File ratTxtFile = new File(buildDirectory, "rat.txt");
86              FileUtils.write(ratTxtFile, "", UTF_8); // Ensure the output file exists and is empty (rerunning the test will append)
87              setVariableValueToObject(mojo, "reportFile", ratTxtFile);
88          }
89          return mojo;
90      }
91  
92      /**
93       * Reads the location of the rat text file from the Mojo.
94       *
95       * @param pMojo The configured Mojo.
96       * @return Value of the "reportFile" property.
97       * @throws Exception An error occurred while reading the property.
98       */
99      private File getRatTxtFile(RatCheckMojo pMojo) throws Exception {
100         return (File) getVariableValueFromObject(pMojo, "reportFile");
101     }

103     private String getDir(RatCheckMojo mojo) {
104         return mojo.getProject().getBasedir().getAbsolutePath().replace("\\","/") + "/";
105     }

183     /**
184      * Tests defining licenses in configuration
185      */
186     public void testIt5() throws Exception {
187         final RatCheckMojo mojo = (RatCheckMojo) newRatMojo("it5", "check", true);
188         final File ratTxtFile = getRatTxtFile(mojo);
189         final String[] expected = { "Notes: 0", "Binaries: 0", "Archives: 0", "Standards: 0$", "Apache Licensed: 0$",
190                 "Generated Documents: 0", "^0 Unknown Licenses" };
191 
192         ReportConfiguration config = mojo.getConfiguration();
193         assertFalse("Should not be adding licenses", config.isAddingLicenses());
194         assertFalse("Should not be forcing licenses", config.isAddingLicensesForced());
195         assertTrue("Should be styling report", config.isStyleReport());
196 
197         ReportConfigurationTest.validateDefaultApprovedLicenses(config, 1);
198         assertTrue(config.getApprovedLicenseCategories().contains(ILicenseFamily.makeCategory("YAL")));
199         ReportConfigurationTest.validateDefaultLicenseFamilies(config, "YAL");
200         assertNotNull(LicenseFamilySetFactory.search("YAL", config.getLicenseFamilies(LicenseFilter.ALL)));
201         ReportConfigurationTest.validateDefaultLicenses(config, "MyLicense", "CpyrT", "RegxT", "SpdxT", "TextT", 
202                 "Not", "All", "Any");
203         assertNotNull(LicenseSetFactory.search("MyLicense", config.getLicenses(LicenseFilter.ALL)));
204         assertNotNull("Should have filesToIgnore", config.getFilesToIgnore());
205         assertThat(config.getFilesToIgnore()).isExactlyInstanceOf(FalseFileFilter.class);
206         assertNotNull("Should have directoriesToIgnore", config.getDirectoriesToIgnore());
207         assertThat(config.getDirectoriesToIgnore()).isExactlyInstanceOf(NameBasedHiddenFileFilter.class);
208         mojo.execute();
209 
210         ensureRatReportIsCorrect(ratTxtFile, expected, TextUtils.EMPTY);
211     }

214      * Runs a check, which should expose no problems.
215      *
216      * @throws Exception The test failed.
217      */
218     public void testRAT_343() throws Exception {
219         final RatCheckMojo mojo = newRatCheckMojo("RAT-343");
220         final File ratTxtFile = getRatTxtFile(mojo);
221         // POM reports AL, BSD and CC BYas BSD because it contains the BSD and CC BY strings
222         final String[] expected = { 
223                 RatTestHelpers.documentOut(false, Document.Type.STANDARD, getDir(mojo) + "pom.xml") +
224                 RatTestHelpers.APACHE_LICENSE +
225                 RatTestHelpers.licenseOut("BSD", "BSD") +
226                 RatTestHelpers.licenseOut("CC BY", "Creative Commons Attribution (Unapproved)"),
227                 "Notes: 0", "Binaries: 0", "Archives: 0",
228                 "Standards: 1$", "Apache Licensed: 1$", "Generated Documents: 0", "^0 Unknown Licenses" };
229 
230         ReportConfiguration config = mojo.getConfiguration();
231         // validate configuration
232         assertThat(config.isAddingLicenses()).isFalse();
233         assertThat(config.isAddingLicensesForced()).isFalse();
234         assertThat(config.getCopyrightMessage()).isNull();
235         assertThat(config.isStyleReport()).isTrue();
236         assertThat(config.getStyleSheet()).withFailMessage("Stylesheet should not be null").isNotNull();
237         assertThat(config.getDirectoriesToIgnore()).withFailMessage("directoriesToIgnore filter should not be null").isNotNull();
238         assertThat(config.getDirectoriesToIgnore()).isExactlyInstanceOf(NameBasedHiddenFileFilter.class);
239         assertThat(config.getFilesToIgnore()).withFailMessage("filesToIgnore filter should not be null").isNotNull();
240         assertThat(config.getFilesToIgnore()).isExactlyInstanceOf(FalseFileFilter.class);
241         
242         ReportConfigurationTest.validateDefaultApprovedLicenses(config, 1);
243         ReportConfigurationTest.validateDefaultLicenseFamilies(config, "BSD", "CC BY");
244         ReportConfigurationTest.validateDefaultLicenses(config, "BSD", "CC BY");
245 
246         mojo.execute();
247         ensureRatReportIsCorrect(ratTxtFile, expected, TextUtils.EMPTY);
248     }

285     /**
286      * Tests verifying gitignore parsing under a special edge case condition
287      * The problem occurs when '/foo.md' is to be ignored and a file with that name exists
288      * in a directory which is the project base directory twice concatenated.
289      * So for this test we must create such a file which is specific for the current
290      * working directory.
291      */
292     public void testRAT362GitIgnore() throws Exception {
293         final RatCheckMojo mojo = newRatCheckMojo("RAT-362-GitIgnore");
294         final File ratTxtFile = getRatTxtFile(mojo);
295         final String dir = getDir(mojo);
296 
297         if (dir.contains(":")) {
298             // The problem this is testing for cannot happen if there is
299             // a Windows drive letter in the name of the directory.
300             // Any duplication of a ':' will make it all fail always.
301             // So there is no point in continuing this test.
302             return;
303         }
304 
305         // Make the target directory for the test file
306         File targetDirectory = new File(dir + dir);
307         if (!targetDirectory.exists()) {
308             assertTrue(targetDirectory.mkdirs());
309         }
310         assertTrue(targetDirectory.isDirectory());
311 
312         // Create the test file with a content on which it must fail
313         File generatedFile = new File(targetDirectory + "/foo.md");
314         final String[] expected = {
315                 "Notes: 0",
316                 "Binaries: 0",
317                 "Archives: 0",
318                 "Standards: 3$",
319                 "Apache Licensed: 2$",
320                 "Generated Documents: 0",
321                 "^1 Unknown Licenses",
322                 RatTestHelpers.documentOut(false, Document.Type.STANDARD, generatedFile.getCanonicalPath()) +
323                     RatTestHelpers.UNKNOWN_LICENSE,
324         };
325         try {
326             BufferedWriter writer = new BufferedWriter(new FileWriter(generatedFile));
327             writer.write("File without a valid license\n");
328             writer.close();
329 
330             mojo.execute();
331             fail("Expected RatCheckException: This check should have failed on the invalid test file");
332         } catch (RatCheckException e) {
333             final String msg = e.getMessage();
334             assertTrue("report filename was not contained in '" + msg + "'", msg.contains(ratTxtFile.getName()));
335             assertFalse("no null allowed in '" + msg + "'", (msg.toUpperCase().contains("NULL")));
336             ensureRatReportIsCorrect(ratTxtFile, expected, TextUtils.EMPTY);
337         } finally {
338             // Cleanup
339             assertTrue(generatedFile.delete());
340         }
341     }
