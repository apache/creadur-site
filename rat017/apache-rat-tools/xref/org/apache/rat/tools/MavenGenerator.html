46  /**
47   * A simple tool to convert CLI options to Maven Mojo base class
48   */
49  public final class MavenGenerator {
50  
51      /** A mapping of external name to internal name if not standard */
52      private static final Map<String, String> RENAME_MAP = new HashMap<>();
53  
54      static {
55          //RENAME_MAP.put("licenses", "config");
56          RENAME_MAP.put("addLicense", "add-license");
57      }
58      /**
59       * List of CLI Options that are not supported by Maven.
60       */
61      private static final List<Option> MAVEN_FILTER_LIST = new ArrayList<>();
62  
63      static {
64          MAVEN_FILTER_LIST.addAll(Arg.DIR.group().getOptions());
65          MAVEN_FILTER_LIST.addAll(Arg.LOG_LEVEL.group().getOptions());
66          MAVEN_FILTER_LIST.add(OptionCollection.HELP);
67      }
68  
69      /**
70       * Filter to remove Options not supported by Maven.
71       */
72      private static final Predicate<Option> MAVEN_FILTER = option -> !(MAVEN_FILTER_LIST.contains(option) || option.getLongOpt() == null);
73  
74      /**
75       * Returns the Option predicate that removes all unsupported Options for the Maven UI.
76       * @return the Option predicate that removes all unsupported Options for the Maven UI.
77       */
78      public static Predicate<Option> getFilter() {
79          return MAVEN_FILTER;
80      }
81  
82      private MavenGenerator() {
83      }
84  
85      /**
86       * Creates the Maven MojoClass
87       * Requires 3 arguments:
88       * <ol>
89       *     <li>the package name for the class</li>
90       *     <li>the simple class name</li>
91       *     <li>the directory in which to write the class file.</li>
92       * </ol>
93       *
94       * @param args the arguments
95       * @throws IOException on error
96       */
97      public static void main(final String[] args) throws IOException {
98          if(args == null || args.length < 3) {
99              System.err.println("At least three arguments are required: package, simple class name, target directory.");
100             return;
101         }
102 
103         String packageName = args[0];
104         String className = args[1];
105         String destDir = args[2];
106         List<MavenOption> options = OptionCollection.buildOptions().getOptions().stream().filter(MAVEN_FILTER)
107                 .map(MavenOption::new).collect(Collectors.toList());
108         String pkgName = String.join(File.separator, new CasedString(StringCase.DOT, packageName).getSegments());
109         File file = new File(new File(new File(destDir), pkgName), className + ".java");
110         System.out.println("Creating " + file);
111         file.getParentFile().mkdirs();
112         try (InputStream template = MavenGenerator.class.getResourceAsStream("/Maven.tpl");
113              FileWriter writer = new FileWriter(file)) {
114             if (template == null) {
115                 throw new RuntimeException("Template /Maven.tpl not found");
116             }
117             LineIterator iter = IOUtils.lineIterator(new InputStreamReader(template, StandardCharsets.UTF_8));
118             while (iter.hasNext()) {
119                 String line = iter.next();
120                 switch (line.trim()) {
121                     case "${static}":
122                         for (Map.Entry<String, String> entry : RENAME_MAP.entrySet()) {
123                             writer.append(format("        xlateName.put(\"%s\", \"%s\");%n", entry.getKey(), entry.getValue()));
124                         }
125                         for (Option option : MAVEN_FILTER_LIST) {
126                             writer.append(format("        unsupportedArgs.add(\"%s\");%n", StringUtils.defaultIfEmpty(option.getLongOpt(), option.getOpt())));
127                         }
128                         break;
129                     case "${methods}":
130                         writeMethods(writer, options);
131                         break;
132                     case "${package}":
133                         writer.append(format("package %s;%n", packageName));
134                         break;
135                     case "${constructor}":
136                         writer.append(format("    protected %s() {}%n", className));
137                         break;
138                     case "${class}":
139                         writer.append(format("public abstract class %s extends AbstractMojo {%n", className));
140                         break;
141                     case "${commonArgs}":
142                         try (InputStream argsTpl = MavenGenerator.class.getResourceAsStream("/Args.tpl")) {
143                             IOUtils.copy(argsTpl, writer, StandardCharsets.UTF_8);
144                         }
145                         break;
146                     default:
147                         writer.append(line).append(System.lineSeparator());
148                         break;
149                 }
150             }
151         }
152     }
153 
154     private static String getComment(final MavenOption option) {
155         StringBuilder sb = new StringBuilder()
156             .append(format("    /**%n     * %s%n     * @param %s the argument.%n", option.getDescription(), option.getName()));
157         if (option.isDeprecated()) {
158             sb.append(format("     * %s%n     * @deprecated%n", option.getDeprecated()));
159         }
160         return sb.append(format("     */%n")).toString();
161     }
162 
163     private static void writeMethods(final FileWriter writer, final List<MavenOption> options) throws IOException {
164         for (MavenOption option : options) {
165             writer.append(getComment(option))
166                     .append(option.getMethodSignature("    ", false)).append(" {").append(System.lineSeparator())
167                     .append(getBody(option))
168                     .append("    }").append(System.lineSeparator());
169             if (option.hasArgs()) {
170                 writer.append(option.getMethodSignature("    ", true)).append(" {").append(System.lineSeparator())
171                         .append(getBody(option))
172                         .append("    }").append(System.lineSeparator());
173             }
174         }
175     }
176 
177     private static String getBody(final MavenOption option) {
178         if (option.hasArg()) {
179             return format("        %sArg(%s, %s);%n", option.hasArgs() ? "add" : "set", option.keyValue(), option.getName());
180         } else {
181             return format("        if (%1$s) {%n            setArg(%2$s, null);%n" +
182                             "        } else {%n            removeArg(%2$s);%n        }%n",
183                     option.getName(), option.keyValue());
184         }
185     }
186 
187     static String createName(final Option option) {
188         String name = option.getLongOpt();
189         name = StringUtils.defaultIfEmpty(RENAME_MAP.get(name), name).toLowerCase(Locale.ROOT);
190         return new CasedString(StringCase.KEBAB, name).toCase(StringCase.CAMEL);
191     }
