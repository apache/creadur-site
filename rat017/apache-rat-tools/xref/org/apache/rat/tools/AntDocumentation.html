54  /**
55   * A simple tool to convert CLI options into an Ant report base class.
56   */
57  public final class AntDocumentation {
58  
59      private final ReportConfiguration config;
60      private final File outputDir;
61  
62      /**
63       * Creates apt documentation fileis for Ant.
64       * Requires 1 argument:
65       * <ol>
66       *     <li>the directory in which to write the documentation files.</li>
67       * </ol>
68       * @param args the arguments.
69       * @throws IOException on error
70       */
71      public static void main(final String[] args) throws IOException {
72          // process the command line to get the output directory.
73          Options opts = OptionCollection.buildOptions();
74          CommandLine commandLine;
75          try {
76              commandLine = DefaultParser.builder().setDeprecatedHandler(DeprecationReporter.getLogReporter())
77                      .setAllowPartialMatching(true).build().parse(opts, args);
78          } catch (ParseException e) {
79              DefaultLog.getInstance().error(e.getMessage());
80              DefaultLog.getInstance().error("Please use the \"--help\" option to see a list of valid commands and options", e);
81              throw new ConfigurationException(e);
82          }
83  
84          String[] remainingArgs = commandLine.getArgs();
85          if (remainingArgs.length == 0) {
86              System.err.println("Output directory must be specified");
87              printUsage(opts);
88              System.exit(1);
89          }
90          File outputDir = new File(remainingArgs[0]);
91          if (outputDir.exists()) {
92              if (!outputDir.isDirectory()) {
93                  DefaultLog.getInstance().error(format("%s is not a directory", remainingArgs[0]));
94                  System.exit(1);
95              }
96          } else {
97              if (!outputDir.mkdirs()) {
98                  DefaultLog.getInstance().error(format("Can not create directory %s", remainingArgs[0]));
99                  System.exit(1);
100             }
101         }
102         // remove any excess arguments and create the configuration.
103         List<String> argsList = new ArrayList<>();
104         argsList.addAll(Arrays.asList(args));
105         argsList.removeAll(Arrays.asList(remainingArgs));
106 
107         ReportConfiguration config = OptionCollection.parseCommands(argsList.toArray(new String[0]), AntDocumentation::printUsage, true);
108         if (config != null) {
109             new AntDocumentation(config, outputDir).execute();
110         }
111     }
112 
113     private static void printUsage(final Options opts) {
114         HelpFormatter f = new HelpFormatter();
115         f.setOptionComparator(OptionCollection.optionComparator);
116         f.setWidth(AbstractHelp.HELP_WIDTH);
117         String header = "\nAvailable options";
118         String footer = "";
119         String cmdLine = format("java -jar apache-rat/target/apache-rat-CURRENT-VERSION.jar %s",
120                 AntDocumentation.class.getName());
121         f.printHelp(cmdLine, header, opts, footer, false);
122         System.exit(0);
123     }
124 
125     private AntDocumentation(ReportConfiguration config, File outputDir) {
126         this.config = config;
127         this.outputDir = outputDir;
128     }
129 
130     public void execute() throws IOException {
131 
132         List<AntOption> options = Arg.getOptions().getOptions().stream().filter(AntGenerator.getFilter()).map(AntOption::new)
133                 .collect(Collectors.toList());
134 
135         writeAttributes(options);
136         writeElements(options);
137         printValueTypes();
138     }
139 
140     public void writeAttributes(List<AntOption> options) {
141         File f = new File(outputDir, "report_attributes.txt");
142         try (Writer out = new OutputStreamWriter(new FileOutputStream(f), StandardCharsets.UTF_8)) {
143             printOptions(out, options, AntOption::isAttribute,
144                     "The attribute value types are listed in a table at the bottom of this page.");
145         } catch (IOException e) {
146             throw new RuntimeException(e);
147         }
148     }
149 
150     public void writeElements(List<AntOption> options) {
151         File f = new File(outputDir, "report_elements.txt");
152         try (Writer out = new OutputStreamWriter(new FileOutputStream(f), StandardCharsets.UTF_8)) {
153 
154             printOptions(out, options, AntOption::isElement,
155                     "The element value types are listed in a table at the bottom of this page.");
156 
157         } catch (IOException e) {
158             throw new RuntimeException(e);
159         }
160     }
161     private void printOptions(Writer out, List<AntOption> options, Predicate<AntOption> typeFilter, String tableCaption) throws IOException {
162         boolean hasDeprecated = options.stream().anyMatch(typeFilter.and(AntOption::isDeprecated));
163 
164         if (hasDeprecated) {
165             AptFormat.writeHeader(out, 2, "Current");
166         }
167 
168         List<List<String>> table = new ArrayList<>();
169         table.add(Arrays.asList("Name", "Description", "Value Type", "Required"));
170         options.stream().filter(typeFilter.and(o -> !o.isDeprecated()))
171                 .map( o -> Arrays.asList( o.getName(), o.getDescription(),
172                         o.hasArg() ? StringUtils.defaultIfEmpty(o.getArgName(), "String") : "boolean",
173                         o.isRequired() ? "true" : "false"))
174                 .forEach(table::add);
175 
176         AptFormat.writeTable(out, table, "*--+--+--+--+", tableCaption);
177 
178         if (hasDeprecated) {
179             AptFormat.writeHeader(out, 2, "Deprecated ");
180 
181             table.clear();
182             table.add(Arrays.asList("Name", "Description", "Argument Type", "Deprecated"));
183 
184             options.stream().filter(typeFilter.and(AntOption::isDeprecated))
185                     .map( o -> Arrays.asList( o.getName(), o.getDescription(),
186                             o.hasArg() ? StringUtils.defaultIfEmpty(o.getArgName(), "String") : "boolean",
187                             o.getDeprecated()))
188                     .forEach(table::add);
189 
190             AptFormat.writeTable(out, table, "*--+--+--+--+", tableCaption);
191         }
192     }
193 
194     private void printValueTypes() throws IOException {
195 
196         File f = new File(outputDir, "report_arg_types.txt");
197         try (Writer writer = new OutputStreamWriter(new FileOutputStream(f), StandardCharsets.UTF_8)) {
198 
199         List<List<String>> table = new ArrayList<>();
200         table.add(Arrays.asList("Value Type", "Description"));
201 
202         for (Map.Entry<String, Supplier<String>> argInfo : OptionCollection.getArgumentTypes().entrySet()) {
203             table.add(Arrays.asList(argInfo.getKey(), argInfo.getValue().get()));
204         }
205 
206         AptFormat.writeTable(writer, table, "*--+--+");
207 
208         } catch (IOException e) {
209             throw new RuntimeException(e);
210         }
211     }

213     private static class AptFormat  {
214 
215         public static void writeLicense(Writer writer) throws IOException {
216             try (InputStream in = AntDocumentation.class.getResourceAsStream("/license.apt")) {
217                 IOUtils.copy(in, writer);
218             }
219         }
220 
221         public static void writeTitle(Writer writer, String title) throws IOException {
222             writer.write(format("        -----%n        %1$s%n        -----%n%n%1$s%n%n", title));
223         }
224 
225         public static void writePara(Writer writer, String paragraph) throws IOException {
226             writer.write(format("  %s%n%n", paragraph));
227         }
228 
229         public static void writeHeader(Writer writer, int level, String text) throws IOException {
230             writer.write(System.lineSeparator());
231             for (int i = 0; i < level; i++) {
232                 writer.write("*");
233             }
234             writer.write(format(" %s%n%n", text));
235         }
236 
237         public static void writeList(Writer writer, Collection<String> list) throws IOException {
238             for (String s : list) {
239                 writer.write(format("    * %s%n", s));
240             }
241             writer.write(System.lineSeparator());
242         }
243 
244         public static void writeTable(Writer writer, Collection<? extends Collection<String>> table, String pattern, String caption) throws IOException {
245             writer.write(format("%s%n", pattern));
246             for (Collection<String> row : table) {
247                 for (String cell : row) {
248                     writer.write(format("| %s ", cell ));
249                 }
250                 writer.write(format("|%n%s%n", pattern));
251             }
252             if (caption != null) {
253                 writer.write(caption);
254             }
255             writer.write(System.lineSeparator());
256         }
257 
258         public static void writeTable(Writer writer, Collection<? extends Collection<String>> table, String pattern) throws IOException {
259             writeTable(writer, table, pattern, null);
260         }
261     }
