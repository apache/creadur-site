50  /**
51   * A simple tool to convert CLI options to Maven and Ant format and produce a CSV file.
52   *
53   * Options
54   * <ul>
55   *     <li>--ant   Produces Ant options in result</li>
56   *     <li>--maven Produces Maven options in result</li>
57   *     <li>--csv   Produces CSV output text is produced</li>
58   * </ul>
59   * Note: if neither --ant nor --maven are included both will be listed.
60   */
61  public final class Naming {
62  
63      private Naming() { }
64  
65      private static final Option WIDTH = Option.builder().longOpt("width").type(Integer.class)
66              .desc("Set the display width of the output").hasArg().build();
67      private static final Option MAVEN = Option.builder().longOpt("maven").desc("Produce Maven name mapping").build();
68      private static final Option ANT = Option.builder().longOpt("ant").desc("Produce Ant name mapping").build();
69      private static final Option CSV = Option.builder().longOpt("csv").desc("Produce CSV format").build();
70      private static final Option CLI = Option.builder().longOpt("cli").desc("Produce CLI name mapping").build();
71      private static final Option INCLUDE_DEPRECATED = Option.builder().longOpt("include-deprecated")
72              .desc("Include deprecated options.").build();
73  
74      private static final Options OPTIONS = new Options().addOption(MAVEN).addOption(ANT).addOption(CLI)
75              .addOption(CSV)
76              .addOption(INCLUDE_DEPRECATED)
77              .addOption(WIDTH);
78      /**
79       * Creates the CSV file.
80       * Requires 1 argument:
81       * <ol>
82       *    <li>the name of the output file with path if desired</li>
83       * </ol>
84       * @throws IOException on error
85       * @param args arguments, only 1 is required.
86       */
87      public static void main(final String[] args) throws IOException, ParseException {
88          if(args == null || args.length < 1) {
89              System.err.println("At least one argument is required: path to file is missing.");
90              return;
91          }
92          CommandLine cl = DefaultParser.builder().build().parse(OPTIONS, args);
93          int width = Math.max(cl.getParsedOptionValue(WIDTH, AbstractHelp.HELP_WIDTH), AbstractHelp.HELP_WIDTH);
94  
95          Predicate<Option> mavenFilter = cl.hasOption(MAVEN) ? MavenGenerator.getFilter() : null;
96  
97          Predicate<Option> antFilter = cl.hasOption(ANT) ? AntGenerator.getFilter() : null;
98          boolean includeDeprecated = cl.hasOption(INCLUDE_DEPRECATED);
99          Predicate<Option> filter = o -> o.hasLongOpt() && (!o.isDeprecated() || includeDeprecated);
100 
101         List<String> columns = new ArrayList<>();
102 
103         if (cl.hasOption(CLI)) {
104             columns.add("CLI");
105         }
106 
107         if (antFilter != null) {
108             columns.add("Ant");
109         }
110 
111         if (mavenFilter != null) {
112             columns.add("Maven");
113         }
114         columns.add("Description");
115         columns.add("Argument Type");
116 
117         try (Writer underWriter = cl.getArgs().length != 0 ? new FileWriter(cl.getArgs()[0]) : new OutputStreamWriter(System.out)) {
118             if (cl.hasOption(CSV)) {
119                 printCSV(columns, filter, cl.hasOption(CLI), mavenFilter, antFilter, underWriter);
120             }
121             else {
122                 printText(columns, filter, cl.hasOption(CLI), mavenFilter, antFilter, underWriter, width);
123             }
124         }
125     }

127     private static List<String> fillColumns(List<String> columns, Option option, boolean addCLI, Predicate<Option> mavenFilter, Predicate<Option> antFilter) {
128         if (addCLI) {
129             if (option.hasLongOpt()) {
130                 columns.add("--" + option.getLongOpt());
131             } else {
132                 columns.add("-" + option.getOpt());
133             }
134         }
135         if (antFilter != null) {
136             columns.add(antFilter.test(option) ? antFunctionName(option) : "-- not supported --");
137         }
138         if (mavenFilter != null) {
139             columns.add(mavenFilter.test(option) ? mavenFunctionName(option) : "-- not supported --");
140         }
141 
142         StringBuilder desc = new StringBuilder();
143         if (option.isDeprecated()) {
144             desc.append("[").append(option.getDeprecated().toString()).append("] ");
145         }
146         columns.add(desc.append(StringUtils.defaultIfEmpty(option.getDescription(),"")).toString());
147         columns.add(option.hasArgName() ? option.getArgName() : option.hasArgs() ? "Strings" : option.hasArg() ? "String" : "-- none --");
148         return columns;
149     }
150 
151     private static void printCSV(List<String> columns, Predicate<Option> filter, boolean addCLI, Predicate<Option> mavenFilter, Predicate<Option> antFilter, Writer underWriter) throws IOException {
152         try (CSVPrinter printer = new CSVPrinter(underWriter, CSVFormat.DEFAULT.builder().setQuoteMode(QuoteMode.ALL).build())) {
153             printer.printRecord(columns);
154             for (Option option : OptionCollection.buildOptions().getOptions()) {
155                 if (filter.test(option)) {
156                     columns.clear();
157                     printer.printRecord(fillColumns(columns, option, addCLI, mavenFilter, antFilter));
158                 }
159             }
160         }
161     }

163     private static int[] calculateColumnWidth(int width, int columnCount, List<List<String>> page) {
164         int[] columnWidth = new int[columnCount];
165         for (List<String> row : page) {
166             for (int i = 0; i < columnCount; i++) {
167                 columnWidth[i] = Math.max(columnWidth[i], row.get(i).length());
168             }
169         }
170         int extra = 0;
171         int averageWidth = (width - ((columnCount-1) * 2)) / columnCount;
172         int[] overage = new int[columnCount];
173         int totalOverage = 0;
174         for (int i = 0; i < columnCount; i++) {
175             if (columnWidth[i] < averageWidth) {
176                 extra += averageWidth - columnWidth[i];
177             } else if (columnWidth[i] > averageWidth) {
178                 overage[i] = columnWidth[i] - averageWidth;
179                 totalOverage += overage[i];
180             }
181         }
182 
183         for (int i = 0; i < columnCount; i++) {
184             if (overage[i] > 0) {
185                 int addl = (int) (extra * overage[i] * 1.0 / totalOverage);
186                 columnWidth[i] = averageWidth + addl;
187             }
188         }
189         return columnWidth;
190     }

192     private static void printText(List<String> columns, Predicate<Option> filter, boolean addCLI, Predicate<Option> mavenFilter,
193                                   Predicate<Option> antFilter, Writer underWriter, int width) throws IOException {
194         List<List<String>> page = new ArrayList<>();
195 
196         int columnCount = columns.size();
197         page.add(columns);
198 
199         for (Option option : OptionCollection.buildOptions().getOptions()) {
200             if (filter.test(option)) {
201                 page.add(fillColumns(new ArrayList<>(), option, addCLI, mavenFilter, antFilter));
202             }
203         }
204         int[] columnWidth = calculateColumnWidth(width, columnCount, page);
205         HelpFormatter helpFormatter;
206         helpFormatter = new HelpFormatter.Builder().get();
207         helpFormatter.setWidth(width);
208 
209 
210         List<Deque<String>> entries = new ArrayList<>();
211         CharArrayWriter cWriter = new CharArrayWriter();
212 
213         // process one line at a time
214         for (List<String> cols : page) {
215             entries.clear();
216             PrintWriter writer = new PrintWriter(cWriter);
217             // print each column into a block of strings.
218             for (int i = 0; i < columnCount; i++) {
219                 String col = cols.get(i);
220                 // split on end of line within a column
221                 for (String line : col.split("\\v")) {
222                     helpFormatter.printWrapped(writer, columnWidth[i], 2, line);
223                 }
224                 writer.flush();
225                 // please the block of strings into a queue.
226                 Deque<String> entryLines = new LinkedList<>();
227                 entryLines.addAll(Arrays.asList(cWriter.toString().split("\\v")));
228                 // put the queue into the entries for this line.
229                 entries.add(entryLines);
230                 cWriter.reset();
231             }
232             // print the entries by printing the items from the queues until all queues are empty.
233             boolean cont = true;
234             while (cont) {
235                 cont = false;
236                 for (int columnNumber = 0; columnNumber < entries.size(); columnNumber++) {
237                     Deque<String> queue = entries.get(columnNumber);
238                     if (queue.isEmpty()) {
239                         underWriter.append(AbstractHelp.createPadding(columnWidth[columnNumber] + 2));
240                     } else {
241                         String ln = queue.pop();
242                         underWriter.append(ln);
243                         underWriter.append(AbstractHelp.createPadding(columnWidth[columnNumber] - ln.length() + 2));
244                         if (!queue.isEmpty()) {
245                             cont = true;
246                         }
247                     }
248                 }
249                 underWriter.append(System.lineSeparator());
250             }
251             underWriter.append(System.lineSeparator());
252         }
253     }

255     public static String mavenFunctionName(final Option option) {
256         MavenOption mavenOption = new MavenOption(option);
257         StringBuilder sb = new StringBuilder();
258         if (mavenOption.isDeprecated()) {
259             sb.append("@Deprecated").append(System.lineSeparator());
260         }
261         return sb.append(format("<%s>", mavenOption.getName())).toString();
262     }
263 
264     private static String antFunctionName(final Option option) {
265         StringBuilder sb = new StringBuilder();
266         AntOption antOption = new AntOption(option);
267         if (option.isDeprecated()) {
268             sb.append("@Deprecated").append(System.lineSeparator());
269         }
270         if (option.hasArgs()) {
271             sb.append(format("<rat:report>%n  <%1$s>text</%1$s>%n</rat:report>", antOption.getName()));
272         } else {
273             sb.append(format("<rat:report %s = 'text'/>", antOption.getName()));
274         }
275         return sb.toString();
276     }
