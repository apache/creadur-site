120     /**
121      * Processes the command line and builds a configuration and executes the
122      * report.
123      * 
124      * @param args the arguments.
125      * @throws Exception on error.
126      */
127     public static void main(String[] args) throws Exception {
128         Options opts = buildOptions();
129 
130         CommandLine cl;
131         try {
132             cl = new DefaultParser().parse(opts, args);
133         } catch (ParseException e) {
134             System.err.println(e.getMessage());
135             System.err.println("Please use the \"--help\" option to see a list of valid commands and options");
136             System.exit(1);
137             return; // dummy return (won't be reached) to avoid Eclipse complaint about possible NPE
138                     // for "cl"
139         }
140 
141         if (cl.hasOption(HELP)) {
142             printUsage(opts);
143         }
144 
145         args = cl.getArgs();
146         if (args == null || args.length != 1) {
147             printUsage(opts);
148         } else {
149             ReportConfiguration configuration = createConfiguration(args[0], cl);
150             configuration.validate(System.err::println);
151 
152             boolean dryRun = false;
153             
154             if (cl.hasOption(LIST_FAMILIES)) {
155                 LicenseFilter f = LicenseFilter.fromText(cl.getOptionValue(LIST_FAMILIES));
156                 if (f != LicenseFilter.none) {                        
157                     dryRun = true;
158                     Reporter.listLicenseFamilies(configuration, f);
159                 }
160             }
161             if (cl.hasOption(LIST_LICENSES)) {
162                 LicenseFilter f = LicenseFilter.fromText(cl.getOptionValue(LIST_LICENSES));
163                 if (f != LicenseFilter.none) {                        
164                     dryRun = true;
165                     Reporter.listLicenses(configuration, f);
166                 }
167             }
168             
169             if (!dryRun) {
170                 Reporter.report(configuration);
171             }
172         }
173     }

175     static ReportConfiguration createConfiguration(String baseDirectory, CommandLine cl) throws IOException {
176         final ReportConfiguration configuration = new ReportConfiguration(DefaultLog.INSTANCE);
177 
178         if (cl.hasOption('o')) {
179             configuration.setOut(new File(cl.getOptionValue('o')));
180         }
181 
182         if (cl.hasOption(SCAN_HIDDEN_DIRECTORIES)) {
183             configuration.setDirectoryFilter(null);
184         }
185 
186         if (cl.hasOption('a') || cl.hasOption('A')) {
187             configuration.setAddLicenseHeaders(cl.hasOption('f') ? AddLicenseHeaders.FORCED : AddLicenseHeaders.TRUE);
188             configuration.setCopyrightMessage(cl.getOptionValue("c"));
189         }
190 
191         if (cl.hasOption(EXCLUDE_CLI)) {
192             String[] excludes = cl.getOptionValues(EXCLUDE_CLI);
193             if (excludes != null) {
194                 final FilenameFilter filter = parseExclusions(Arrays.asList(excludes));
195                 configuration.setInputFileFilter(filter);
196             }
197         } else if (cl.hasOption(EXCLUDE_FILE_CLI)) {
198             String excludeFileName = cl.getOptionValue(EXCLUDE_FILE_CLI);
199             if (excludeFileName != null) {
200                 final FilenameFilter filter = parseExclusions(
201                         FileUtils.readLines(new File(excludeFileName), StandardCharsets.UTF_8));
202                 configuration.setInputFileFilter(filter);
203             }
204         }
205 
206         if (cl.hasOption(XML)) {
207             configuration.setStyleReport(false);
208         } else {
209             configuration.setStyleReport(true);
210             if (cl.hasOption(STYLESHEET_CLI)) {
211                 String[] style = cl.getOptionValues(STYLESHEET_CLI);
212                 if (style.length != 1) {
213                     System.err.println("Please specify a single stylesheet");
214                     System.exit(1);
215                 }
216                 configuration.setStyleSheet(() -> Files.newInputStream(Paths.get(style[0])));
217             }
218         }
219 
220         Defaults.Builder defaultBuilder = Defaults.builder();
221         if (cl.hasOption(NO_DEFAULTS)) {
222             defaultBuilder.noDefault();
223         }
224         if (cl.hasOption(LICENSES)) {
225             for (String fn : cl.getOptionValues(LICENSES)) {
226                 defaultBuilder.add(fn);
227             }
228         }
229         Defaults defaults = defaultBuilder.build();
230         configuration.setFrom(defaults);
231         configuration.setReportable(getDirectory(baseDirectory, configuration));
232         return configuration;
233     }

235     /**
236      * Creates a filename filter from patterns to exclude.
237      * 
238      * @param excludes the list of patterns to exclude.
239      * @return the FilenameFilter tht excludes the patterns
240      */
241     static FilenameFilter parseExclusions(List<String> excludes) {
242         final OrFileFilter orFilter = new OrFileFilter();
243         int ignoredLines = 0;
244         for (String exclude : excludes) {
245             try {
246                 // skip comments
247                 if (exclude.startsWith("#") || StringUtils.isEmpty(exclude)) {
248                     ignoredLines++;
249                     continue;
250                 }
251 
252                 String exclusion = exclude.trim();
253                 // interpret given patterns as regular expression, direct file names or
254                 // wildcards to give users more choices to configure exclusions
255                 orFilter.addFileFilter(new RegexFileFilter(exclusion));
256                 orFilter.addFileFilter(new NameFileFilter(exclusion));
257                 orFilter.addFileFilter(WildcardFileFilter.builder().setWildcards(exclusion).get());
258             } catch (PatternSyntaxException e) {
259                 System.err.println("Will skip given exclusion '" + exclude + "' due to " + e);
260             }
261         }
262         System.err.println("Ignored " + ignoredLines + " lines in your exclusion files as comments or empty lines.");
263         return new NotFileFilter(orFilter);
264     }

266     static Options buildOptions() {
267         String licFilterValues = String.join(", ", 
268                 Arrays.stream(LicenseFilter.values()).map(LicenseFilter::name).collect(Collectors.toList()));
269 
270         Options opts = new Options()
271         
272         .addOption(
273                 Option.builder().hasArg(true).longOpt(LIST_FAMILIES)
274                 .desc("List the defined license families (default is none). Valid options are: "+licFilterValues+".")
275                 .build())
276         .addOption(
277                 Option.builder().hasArg(true).longOpt(LIST_LICENSES)
278                 .desc("List the defined licenses (default is none). Valid options are: "+licFilterValues+".")
279                 .build())
280 
281         .addOption(new Option(HELP, "help", false, "Print help for the RAT command line interface and exit."));
282         
283 
284         Option out = new Option("o", "out", true,
285                 "Define the output file where to write a report to (default is System.out).");
286         opts.addOption(out);
287 
288         String defaultHandlingText = " By default all approved default licenses are used";
289         Option noDefaults = new Option(null, NO_DEFAULTS, false, "Ignore default configuration." + defaultHandlingText);
290         opts.addOption(noDefaults);
291 
292         opts.addOption(null, LICENSES, true, "File names or URLs for license definitions");
293         opts.addOption(null, SCAN_HIDDEN_DIRECTORIES, false, "Scan hidden directories");
294 
295         OptionGroup addLicenseGroup = new OptionGroup();
296         String addLicenseDesc = "Add the default license header to any file with an unknown license that is not in the exclusion list. "
297                 + "By default new files will be created with the license header, "
298                 + "to force the modification of existing files use the --force option.";
299 
300         // RAT-85/RAT-203: Deprecated! added only for convenience and for backwards
301         // compatibility
302         Option addLicence = new Option(ADD_OLD, "addLicence", false, addLicenseDesc);
303         addLicenseGroup.addOption(addLicence);
304         Option addLicense = new Option(ADD, "addLicense", false, addLicenseDesc);
305         addLicenseGroup.addOption(addLicense);
306         opts.addOptionGroup(addLicenseGroup);
307 
308         Option write = new Option(FORCE, "force", false,
309                 "Forces any changes in files to be written directly to the source files (i.e. new files are not created).");
310         opts.addOption(write);
311 
312         Option copyright = new Option(COPYRIGHT, "copyright", true,
313                 "The copyright message to use in the license headers, usually in the form of \"Copyright 2008 Foo\"");
314         opts.addOption(copyright);
315 
316         final Option exclude = Option.builder(EXCLUDE_CLI).argName("expression").longOpt("exclude").hasArgs()
317                 .desc("Excludes files matching wildcard <expression>. "
318                         + "Note that --dir is required when using this parameter. " + "Allows multiple arguments.")
319                 .build();
320         opts.addOption(exclude);
321 
322         final Option excludeFile = Option.builder(EXCLUDE_FILE_CLI).argName("fileName").longOpt("exclude-file")
323                 .hasArgs().desc("Excludes files matching regular expression in <file> "
324                         + "Note that --dir is required when using this parameter. ")
325                 .build();
326         opts.addOption(excludeFile);
327 
328         Option dir = new Option("d", "dir", false, "Used to indicate source when using --exclude");
329         opts.addOption(dir);
330 
331         OptionGroup outputType = new OptionGroup();
332 
333         Option xml = new Option(XML, "xml", false, "Output the report in raw XML format.  Not compatible with -s");
334         outputType.addOption(xml);
335 
336         Option xslt = new Option(STYLESHEET_CLI, "stylesheet", true,
337                 "XSLT stylesheet to use when creating the" + " report.  Not compatible with -x");
338         outputType.addOption(xslt);
339         opts.addOptionGroup(outputType);
340 
341         return opts;
342     }

344     private static void printUsage(Options opts) {
345         HelpFormatter f = new HelpFormatter();
346         f.setOptionComparator(new OptionComparator());
347         String header = "\nAvailable options";
348 
349         String footer = "\nNOTE:\n" + "Rat is really little more than a grep ATM\n"
350                 + "Rat is also rather memory hungry ATM\n" + "Rat is very basic ATM\n"
351                 + "Rat highlights possible issues\n" + "Rat reports require interpretation\n"
352                 + "Rat often requires some tuning before it runs well against a project\n"
353                 + "Rat relies on heuristics: it may miss issues\n";
354 
355         f.printHelp("java -jar apache-rat/target/apache-rat-CURRENT-VERSION.jar [options] [DIR|TARBALL]", header, opts,
356                 footer, false);
357         System.exit(0);
358     }

364     /**
365      * Creates an IReportable object from the directory name and ReportConfiguration
366      * object.
367      * 
368      * @param baseDirectory the directory that contains the files to report on.
369      * @param config the ReportConfiguration.
370      * @return the IReportable instance containing the files.
371      */
372     private static IReportable getDirectory(String baseDirectory, ReportConfiguration config) {
373         try (PrintStream out = new PrintStream(config.getOutput().get())) {
374             File base = new File(baseDirectory);
375             
376             if (!base.exists()) {
377                 config.getLog().log(Level.ERROR, "Directory '"+baseDirectory+"' does not exist");
378                 return null;
379             }
380 
381             if (base.isDirectory()) {
382                 return new DirectoryWalker(base, config.getInputFileFilter(), config.getDirectoryFilter());
383             }
384 
385             try {
386                 return new ArchiveWalker(base, config.getInputFileFilter());
387             } catch (IOException ex) {
388                 config.getLog().log(Level.ERROR, "file '"+baseDirectory+"' is not valid gzip data.");
389                 return null;
390             }
391         } catch (IOException e) {
392             throw new ConfigurationException("Error opening output", e);
393         }
394     }

396     /**
397      * This class implements the {@code Comparator} interface for comparing Options.
398      */
399     private static class OptionComparator implements Comparator<Option>, Serializable {
400         /** The serial version UID. */
401         private static final long serialVersionUID = 5305467873966684014L;
402 
403         private String getKey(Option opt) {
404             String key = opt.getOpt();
405             key = key == null ? opt.getLongOpt() : key;
406             return key;
407         }
408 
409         /**
410          * Compares its two arguments for order. Returns a negative integer, zero, or a
411          * positive integer as the first argument is less than, equal to, or greater
412          * than the second.
413          *
414          * @param opt1 The first Option to be compared.
415          * @param opt2 The second Option to be compared.
416          * @return a negative integer, zero, or a positive integer as the first argument
417          * is less than, equal to, or greater than the second.
418          */
419         @Override
420         public int compare(final Option opt1, final Option opt2) {
421             return getKey(opt1).compareToIgnoreCase(getKey(opt2));
422         }
423     }
