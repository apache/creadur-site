127     /**
128      * This method is called when the report generation is invoked directly as a
129      * standalone Mojo.
130      *
131      * @throws MojoExecutionException if an error occurs when generating the report
132      * @see org.apache.maven.plugin.Mojo#execute()
133      */
134     @Override
135     public void execute() throws MojoExecutionException {
136         if (!canGenerateReport()) {
137             return;
138         }
139 
140         File outputDirectory = new File(getOutputDirectory());
141 
142         String filename = getOutputName() + ".html";
143 
144         Locale locale = Locale.getDefault();
145 
146         try {
147             SiteRenderingContext siteContext = createSiteRenderingContext(locale);
148 
149             // copy resources
150             getSiteRenderer().copyResources(siteContext, outputDirectory);
151 
152             // TODO Replace null with real value
153             RenderingContext docRenderingContext = new RenderingContext(outputDirectory, filename, null);
154 
155             SiteRendererSink sink = new SiteRendererSink(docRenderingContext);
156 
157             generate(sink, null, locale);
158 
159             // MSHARED-204: only render Doxia sink if not an external report
160             if (!isExternalReport()) {
161                 outputDirectory.mkdirs();
162 
163                 try (Writer writer = new OutputStreamWriter(
164                         Files.newOutputStream(new File(outputDirectory, filename).toPath()), getOutputEncoding())) {
165                     // render report
166                     getSiteRenderer().mergeDocumentIntoSite(writer, sink, siteContext);
167                 }
168             }
169 
170             // copy generated resources also
171             getSiteRenderer().copyResources(siteContext, outputDirectory);
172         } catch (RendererException | IOException | MavenReportException e) {
173             throw new MojoExecutionException(
174                     "An error has occurred in " + getName(Locale.ENGLISH) + " report generation.", e);
175         }
176     }

178     private SiteRenderingContext createSiteRenderingContext(Locale locale) throws MavenReportException, IOException {
179         DecorationModel decorationModel = new DecorationModel();
180 
181         Map<String, Object> templateProperties = new HashMap<>();
182         // We tell the skin that we are rendering in standalone mode
183         templateProperties.put("standalone", Boolean.TRUE);
184         templateProperties.put("project", getProject());
185         templateProperties.put("inputEncoding", getInputEncoding());
186         templateProperties.put("outputEncoding", getOutputEncoding());
187         // Put any of the properties in directly into the Velocity context
188         for (Map.Entry<Object, Object> entry : getProject().getProperties().entrySet()) {
189             templateProperties.put((String) entry.getKey(), entry.getValue());
190         }
191 
192         SiteRenderingContext context;
193         try {
194             Artifact skinArtifact = siteTool.getSkinArtifactFromRepository(session.getLocalRepository(),
195                     remoteRepositories, decorationModel);
196 
197             getLog().debug(
198                     buffer().a("Rendering content with ").strong(skinArtifact.getId() + " skin").a('.').build());
199 
200             context = siteRenderer.createContextForSkin(skinArtifact, templateProperties, decorationModel,
201                     project.getName(), locale);
202         } catch (SiteToolException e) {
203             throw new MavenReportException("Failed to retrieve skin artifact", e);
204         } catch (RendererException e) {
205             throw new MavenReportException("Failed to create context for skin", e);
206         }
207 
208         // Generate static site
209         context.setRootDirectory(project.getBasedir());
210 
211         return context;
212     }

241     /**
242      * This method is called when the report generation is invoked by
243      * maven-site-plugin.
244      *
245      * @param sink the sink to use for the generation.
246      * @param sinkFactory the sink factory to use for the generation.
247      * @param locale the wanted locale to generate the report, could be null.
248      * @throws MavenReportException if any
249      */
250     @Override
251     public void generate(Sink sink, SinkFactory sinkFactory, Locale locale) throws MavenReportException {
252         if (!canGenerateReport()) {
253             getLog().info("This report cannot be generated as part of the current build. "
254                     + "The report name should be referenced in this line of output.");
255             return;
256         }
257 
258         this.sink = sink;
259 
260         this.sinkFactory = sinkFactory;
261 
262         executeReport(locale);
263 
264         closeReport();
265     }

352     /**
353      * Searches for a Rat artifact in the dependency list and returns its version.
354      *
355      * @return Version number, if found, or null.
356      */
357     // TODO The canonical way is to read the pom.properties for the artifact desired
358     // in META-INF/maven
359     private String getRatVersion() {
360         // noinspection unchecked
361         for (Artifact a : getProject().getDependencyArtifacts()) {
362             if ("rat-lib".equals(a.getArtifactId())) {
363                 return a.getVersion();
364             }
365         }
366         return null;
367     }

369     /**
370      * Writes the report to the Doxia sink.
371      *
372      * @param locale The locale to use for writing the report.
373      * @throws MavenReportException Writing the report failed.
374      */
375     protected void executeReport(Locale locale) throws MavenReportException {
376         ResourceBundle bundle = getBundle(locale);
377         final String title = bundle.getString("report.rat.title");
378         sink.head();
379         sink.title();
380         sink.text(title);
381         sink.title_();
382         sink.head_();
383 
384         sink.body();
385 
386         sink.section1();
387         sink.sectionTitle1();
388         sink.text(title);
389         sink.sectionTitle1_();
390 
391         sink.paragraph();
392         sink.text(bundle.getString("report.rat.link") + " ");
393         sink.link(bundle.getString("report.rat.url"));
394         sink.text(bundle.getString("report.rat.fullName"));
395         sink.link_();
396         final String ratVersion = getRatVersion();
397         if (ratVersion != null) {
398             sink.text(" " + ratVersion);
399         }
400         sink.text(".");
401         sink.paragraph_();
402 
403         sink.paragraph();
404         sink.verbatim(SinkEventAttributeSet.BOXED);
405         try {
406             ReportConfiguration config = getConfiguration();
407             config.setFrom(getDefaultsBuilder().build());
408             //config.setStyleSheet(Defaults.getUnapprovedLicensesStyleSheet());
409             logLicenses(config.getLicenses(LicenseFilter.all));
410             ByteArrayOutputStream baos = new ByteArrayOutputStream();
411             config.setOut(()->baos);
412             Reporter.report(config);
413             sink.text(baos.toString());
414         } catch (Exception e) {
415             throw new MavenReportException(e.getMessage(), e);
416         }
417         sink.verbatim_();
418         sink.paragraph_();
419         sink.section1_();
420         sink.body_();
421     }
