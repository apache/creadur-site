44      private static final String EXCLUDE_CLI = "e";
45      private static final String EXCLUDE_FILE_CLI = "E";
46      private static final String STYLESHEET_CLI = "s";
47      private static final String HELP = "h";

49      public static final void main(String[] args) throws Exception {
50          final ReportConfiguration configuration = new ReportConfiguration();
51          configuration.setHeaderMatcher(Defaults.createDefaultMatcher());
52          configuration.setApproveDefaultLicenses(true);
53          Options opts = buildOptions();
54  
55          CommandLine cl = null;
56          try {
57              cl = new DefaultParser().parse(opts, args);
58          } catch (ParseException e) {
59              System.err.println("Please use the \"--help\" option to see a list of valid commands and options");
60              System.exit(1);
61              return; // dummy return (won't be reached) to avoid Eclipse complaint about possible NPE for "cl"
62          }
63  
64          if (cl.hasOption(HELP)) {
65              printUsage(opts);
66          }
67  
68          args = cl.getArgs();
69          if (args == null || args.length != 1) {
70              printUsage(opts);
71          } else {
72              Report report = new Report(args[0]);
73  
74              if (cl.hasOption('a') || cl.hasOption('A')) {
75                  configuration.setAddingLicenses(true);
76                  configuration.setAddingLicensesForced(cl.hasOption('f'));
77                  configuration.setCopyrightMessage(cl.getOptionValue("c"));
78              }
79  
80              if (cl.hasOption(EXCLUDE_CLI)) {
81                  String[] excludes = cl.getOptionValues(EXCLUDE_CLI);
82                  if (excludes != null) {
83                      final FilenameFilter filter = parseExclusions(Arrays.asList(excludes));
84                      report.setInputFileFilter(filter);
85                  }
86              } else if (cl.hasOption(EXCLUDE_FILE_CLI)) {
87                  String excludeFileName = cl.getOptionValue(EXCLUDE_FILE_CLI);
88                  if (excludeFileName != null) {
89                      final FilenameFilter filter = parseExclusions(FileUtils.readLines(new File(excludeFileName), Charset.forName("UTF-8")));
90                      report.setInputFileFilter(filter);
91                  }
92              }
93              if (cl.hasOption('x')) {
94                  report.report(System.out, configuration);
95              } else {
96                  if (!cl.hasOption(STYLESHEET_CLI)) {
97                      report.styleReport(System.out, configuration);
98                  } else {
99                      String[] style = cl.getOptionValues(STYLESHEET_CLI);
100                     if (style.length != 1) {
101                         System.err.println("please specify a single stylesheet");
102                         System.exit(1);
103                     }
104                     try {
105                         report(System.out,
106                                 report.getDirectory(System.out),
107                                 new FileInputStream(style[0]),
108                                 configuration);
109                     } catch (FileNotFoundException fnfe) {
110                         System.err.println("stylesheet " + style[0]
111                                 + " doesn't exist");
112                         System.exit(1);
113                     }
114                 }
115             }
116         }
117     }

119     static FilenameFilter parseExclusions(List<String> excludes) throws IOException {
120         final OrFileFilter orFilter = new OrFileFilter();
121         int ignoredLines = 0;
122         for (String exclude : excludes) {
123             try {
124                 // skip comments
125                 if(exclude.startsWith("#") || StringUtils.isEmpty(exclude)) {
126                     ignoredLines++;
127                     continue;
128                 }
129 
130                 String exclusion = exclude.trim();
131                 // interpret given patterns as regular expression, direct file names or wildcards to give users more choices to configure exclusions
132                 orFilter.addFileFilter(new RegexFileFilter(exclusion));
133                 orFilter.addFileFilter(new NameFileFilter(exclusion));
134                 orFilter.addFileFilter(new WildcardFileFilter(exclusion));
135             } catch(PatternSyntaxException e) {
136                 System.err.println("Will skip given exclusion '" + exclude + "' due to " + e);
137             }
138         }
139         System.err.println("Ignored " + ignoredLines + " lines in your exclusion files as comments or empty lines.");
140         return new NotFileFilter(orFilter);
141     }

143     private static Options buildOptions() {
144         Options opts = new Options();
145 
146         Option help = new Option(HELP, "help", false,
147                 "Print help for the RAT command line interface and exit");
148         opts.addOption(help);
149 
150         OptionGroup addLicenseGroup = new OptionGroup();
151         String addLicenseDesc = "Add the default license header to any file with an unknown license that is not in the exclusion list. " +
152                 "By default new files will be created with the license header, " +
153                 "to force the modification of existing files use the --force option.";
154 
155         // RAT-85/RAT-203: Deprecated! added only for convenience and for backwards compatibility
156         Option addLicence = new Option(
157                 "a",
158                 "addLicence",
159                 false,
160                 addLicenseDesc);
161         addLicenseGroup.addOption(addLicence);
162         Option addLicense = new Option(
163                 "A",
164                 "addLicense",
165                 false,
166                 addLicenseDesc);
167         addLicenseGroup.addOption(addLicense);
168         opts.addOptionGroup(addLicenseGroup);
169 
170         Option write = new Option(
171                 "f",
172                 "force",
173                 false,
174                 "Forces any changes in files to be written directly to the source files (i.e. new files are not created)");
175         opts.addOption(write);
176 
177         Option copyright = new Option(
178                 "c",
179                 "copyright",
180                 true,
181                 "The copyright message to use in the license headers, usually in the form of \"Copyright 2008 Foo\"");
182         opts.addOption(copyright);
183 
184         final Option exclude = Option.builder(EXCLUDE_CLI)
185                 .argName("expression")
186                 .longOpt("exclude")
187                 .hasArgs()
188                 .desc("Excludes files matching wildcard <expression>. " +
189                         "Note that --dir is required when using this parameter. " +
190                         "Allows multiple arguments.")
191                 .build();
192         opts.addOption(exclude);
193 
194         final Option excludeFile = Option.builder(EXCLUDE_FILE_CLI)
195                 .argName("fileName")
196                 .longOpt("exclude-file")
197                 .hasArgs()
198                 .desc("Excludes files matching regular expression in <file> " +
199                         "Note that --dir is required when using this parameter. ")
200                 .build();
201         opts.addOption(excludeFile);
202 
203         Option dir = new Option(
204                 "d",
205                 "dir",
206                 false,
207                 "Used to indicate source when using --exclude");
208         opts.addOption(dir);
209 
210         OptionGroup outputType = new OptionGroup();
211 
212         Option xml = new Option(
213                 "x",
214                 "xml",
215                 false,
216                 "Output the report in raw XML format.  Not compatible with -s");
217         outputType.addOption(xml);
218 
219         Option xslt = new Option(STYLESHEET_CLI,
220                 "stylesheet",
221                 true,
222                 "XSLT stylesheet to use when creating the"
223                         + " report.  Not compatible with -x");
224         outputType.addOption(xslt);
225         opts.addOptionGroup(outputType);
226 
227         return opts;
228     }

230     private static void printUsage(Options opts) {
231         HelpFormatter f = new HelpFormatter();
232         String header = "\nAvailable options";
233 
234         String footer = "\nNOTE:\n" +
235                 "Rat is really little more than a grep ATM\n" +
236                 "Rat is also rather memory hungry ATM\n" +
237                 "Rat is very basic ATM\n" +
238                 "Rat highlights possible issues\n" +
239                 "Rat reports require interpretation\n" +
240                 "Rat often requires some tuning before it runs well against a project\n" +
241                 "Rat relies on heuristics: it may miss issues\n";
242 
243         f.printHelp("java -jar apache-rat/target/apache-rat-CURRENT-VERSION.jar [options] [DIR|TARBALL]",
244                 header, opts, footer, false);
245         System.exit(0);
246     }

279     /**
280      * @param out           - the output stream to receive the styled report
281      * @param configuration - current configuration options.
282      * @return the currently collected numerical statistics.
283      * @throws Exception in case of errors.
284      * @since Rat 0.8
285      */
286     public ClaimStatistic report(PrintStream out,
287                                  ReportConfiguration configuration)
288             throws Exception {
289         final IReportable base = getDirectory(out);
290         if (base != null) {
291             return report(base, new OutputStreamWriter(out), configuration);
292         }
293         return null;
294     }
295 
296     private IReportable getDirectory(PrintStream out) {
297         File base = new File(baseDirectory);
298         if (!base.exists()) {
299             out.print("ERROR: ");
300             out.print(baseDirectory);
301             out.print(" does not exist.\n");
302             return null;
303         }
304 
305         if (base.isDirectory()) {
306             return new DirectoryWalker(base, inputFileFilter);
307         }
308 
309         try {
310             return new ArchiveWalker(base, inputFileFilter);
311         } catch (IOException ex) {
312             out.print("ERROR: ");
313             out.print(baseDirectory);
314             out.print(" is not valid gzip data.\n");
315             return null;
316         }
317     }

354     /**
355      * Output a report that is styled using a defined stylesheet.
356      *
357      * @param out            the stream to write the report to
358      * @param base           the files or directories to report on
359      * @param style          an input stream representing the stylesheet to use for styling the report
360      * @param pConfiguration current report configuration.
361      * @throws IOException                       in case of I/O errors.
362      * @throws TransformerConfigurationException in case of XML errors.
363      * @throws InterruptedException              in case of threading errors.
364      * @throws RatException                      in case of internal errors.
365      */
366     public static void report(PrintStream out, IReportable base, final InputStream style,
367                               ReportConfiguration pConfiguration)
368             throws IOException, TransformerConfigurationException, InterruptedException, RatException {
369         report(new OutputStreamWriter(out), base, style, pConfiguration);
370     }

372     /**
373      * Output a report that is styled using a defined stylesheet.
374      *
375      * @param out            the writer to write the report to
376      * @param base           the files or directories to report on
377      * @param style          an input stream representing the stylesheet to use for styling the report
378      * @param pConfiguration current report configuration.
379      * @return the currently collected numerical statistics.
380      * @throws IOException                       in case of I/O errors.
381      * @throws TransformerConfigurationException in case of XML errors.
382      * @throws InterruptedException              in case of threading errors.
383      * @throws RatException                      in case of internal errors.
384      */
385     public static ClaimStatistic report(Writer out, IReportable base, final InputStream style,
386                                         ReportConfiguration pConfiguration)
387             throws IOException, TransformerConfigurationException, InterruptedException, RatException {
388         PipedReader reader = new PipedReader();
389         PipedWriter writer = new PipedWriter(reader);
390         ReportTransformer transformer = new ReportTransformer(out, style, reader);
391         Thread transformerThread = new Thread(transformer);
392         transformerThread.start();
393         final ClaimStatistic statistic = report(base, writer, pConfiguration);
394         writer.flush();
395         writer.close();
396         transformerThread.join();
397         return statistic;
398     }

400     /**
401      * @param container      the files or directories to report on
402      * @param out            the writer to write the report to
403      * @param pConfiguration current report configuration.
404      * @return the currently collected numerical statistics.
405      * @throws IOException  in case of I/O errors.
406      * @throws RatException in case of internal errors.
407      */
408     public static ClaimStatistic report(final IReportable container, final Writer out,
409                                         ReportConfiguration pConfiguration) throws IOException, RatException {
410         IXmlWriter writer = new XmlWriter(out);
411         final ClaimStatistic statistic = new ClaimStatistic();
412         RatReport report = XmlReportFactory.createStandardReport(writer, statistic, pConfiguration);
413         report.startReport();
414         container.run(report);
415         report.endReport();
416         writer.closeDocument();
417         return statistic;
418     }
