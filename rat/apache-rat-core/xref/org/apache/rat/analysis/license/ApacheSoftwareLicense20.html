19  package org.apache.rat.analysis.license;
20  
21  import org.apache.rat.api.MetaData;
22  
23  /**
24   * Matches Apache License, Version 2.0
25   *
26   */
27  public final class ApacheSoftwareLicense20 extends SimplePatternBasedLicense {
28      public static final String FIRST_LICENSE_LINE = "Licensed under the Apache License, Version 2.0 (the \"License\")";
29      public static final String FIRST_LICENSE_LINE_SHORT = "Licensed to the Apache Software Foundation (ASF) under one or more contributor license agreements; and to You under the Apache License, Version 2.0.";
30      public static final String LICENSE_REFERENCE_LINE = "http://www.apache.org/licenses/LICENSE-2.0";
31  
32      // These are all allowed variants as mentioned in https://issues.apache.org/jira/browse/LEGAL-265
33      public static final String LICENSE_URL_HTTP       = LICENSE_REFERENCE_LINE;
34      public static final String LICENSE_URL_HTTPS      = "https://www.apache.org/licenses/LICENSE-2.0";
35      public static final String LICENSE_URL_HTTP_HTML  = "http://www.apache.org/licenses/LICENSE-2.0.html";
36      public static final String LICENSE_URL_HTTPS_HTML = "https://www.apache.org/licenses/LICENSE-2.0.html";
37      public static final String LICENSE_URL_HTTP_TXT   = "http://www.apache.org/licenses/LICENSE-2.0.txt";
38      public static final String LICENSE_URL_HTTPS_TXT  = "https://www.apache.org/licenses/LICENSE-2.0.txt";
39      public ApacheSoftwareLicense20() {
40          super(MetaData.RAT_LICENSE_FAMILY_CATEGORY_DATUM_ASL, MetaData.RAT_LICENSE_FAMILY_NAME_DATUM_APACHE_LICENSE_VERSION_2_0,
41                  "", new String[]{FIRST_LICENSE_LINE, FIRST_LICENSE_LINE_SHORT,
42                          LICENSE_URL_HTTP,       LICENSE_URL_HTTPS,
43                          LICENSE_URL_HTTP_HTML,  LICENSE_URL_HTTPS_HTML,
44                          LICENSE_URL_HTTP_TXT,   LICENSE_URL_HTTPS_TXT,});
45      }
46  }
