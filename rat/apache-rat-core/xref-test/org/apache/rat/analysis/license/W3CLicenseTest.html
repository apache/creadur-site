19  package org.apache.rat.analysis.license;
20  
21  import java.util.Arrays;
22  import java.util.Collection;
23  
24  import org.junit.runner.RunWith;
25  import org.junit.runners.Parameterized;
26  import org.junit.runners.Parameterized.Parameters;
27  
28  @RunWith(Parameterized.class)
29  public class W3CLicenseTest extends AbstractLicenseTest {
30  
31      private static String W3C_note = "Note that W3C requires a NOTICE.\n" + "All modifications require notes.\n"
32              + "See http://www.w3.org/Consortium/Legal/2002/copyright-software-20021231.";
33      private static Object[] W3C = { "W3C", "W3C Software Copyright", W3C_note,
34              new String[][] { { "fulltext", "http://www.w3.org/Consortium/Legal/2002/copyright-software-20021231" },
35                      { "spdx-tab", "SPDX-License-Identifier:\tW3C" },
36                      { "spdx-space", "SPDX-License-Identifier: W3C" }, } };
37  
38      private static String W3CD_note = "Note that W3CD does not allow modifications.\n"
39              + "See http://www.w3.org/Consortium/Legal/2002/copyright-documents-20021231.";
40      private static Object[] W3CD = { "W3CD", "W3C Document Copyright", W3CD_note, new String[][] {
41              { "fulltext", "http://www.w3.org/Consortium/Legal/2002/copyright-documents-20021231" }, } };
42  
43      @Parameters(name = "{0}")
44      public static Collection<Object[]> data() {
45          return Arrays.asList(W3C, W3CD);
46      }
47  
48      public W3CLicenseTest(String id, String name, String note, String[][] targets) {
49          super(id, id, name, note, targets);
50      }
51  
52  }
