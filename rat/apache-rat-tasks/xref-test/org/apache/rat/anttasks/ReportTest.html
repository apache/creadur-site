53      @Test
54      public void testWithReportSentToFile() throws Exception {
55          final File reportFile = new File(getTempDir(), "selftest.report");
56          if (!getTempDir().mkdirs() && !getTempDir().isDirectory()) {
57              throw new IOException("Could not create temporary directory " + getTempDir());
58          }
59          final String alLine = "AL +\\Q" + getAntFileName() + "\\E";
60          if (reportFile.isFile() && !reportFile.delete()) {
61              throw new IOException("Unable to remove report file " + reportFile);
62          }
63          buildRule.executeTarget("testWithReportSentToFile");
64          assertLogDoesNotMatch(alLine);
65          assertTrue("Expected report file " + reportFile, reportFile.isFile());
66          assertFileMatches(reportFile, alLine);
67      }

111     private String getFirstLine(File pFile) throws IOException {
112         FileInputStream fis = null;
113         InputStreamReader reader = null;
114         BufferedReader breader = null;
115         try {
116             fis = new FileInputStream(pFile);
117             reader = new InputStreamReader(fis, StandardCharsets.UTF_8);
118             breader = new BufferedReader(reader);
119             final String result = breader.readLine();
120             breader.close();
121             return result;
122         } finally {
123             IOUtils.closeQuietly(fis);
124             IOUtils.closeQuietly(reader);
125             IOUtils.closeQuietly(breader);
126         }
127     }

143     /**
144      * Test correct generation of string result if non-UTF8 file.encoding is set.
145      */
146     @Test
147     public void testISO88591() throws Exception {
148     	// In previous versions of the JDK, it used to be possible to
149     	// change the value of file.encoding at runtime. As of Java 16,
150     	// this is no longer possible. Instead, at this point, we check,
151     	// that file.encoding is actually ISO-8859-1. (Within Maven, this
152     	// is enforced by the configuration of the surefire plugin.) If not,
153     	// we skip this test.
154     	Assume.assumeTrue("Expected file.encoding=ISO-8859-1", "ISO-8859-1".equals(System.getProperty("file.encoding")));
155         buildRule.executeTarget("testISO88591");
156         assertTrue("Log should contain the test umlauts", buildRule.getLog().contains("\u00E4\u00F6\u00FC\u00C4\u00D6\u00DC\u00DF"));
157     }

159     /**
160      * Test correct generation of XML file if non-UTF8 file.encoding is set.
161      */
162     @Test
163     public void testISO88591WithFile() throws Exception {
164     	// In previous versions of the JDK, it used to be possible to
165     	// change the value of file.encoding at runtime. As of Java 16,
166     	// this is no longer possible. Instead, at this point, we check,
167     	// that file.encoding is actually ISO-8859-1. (Within Maven, this
168     	// is enforced by the configuration of the surefire plugin.) If not,
169     	// we skip this test.
170     	Assume.assumeTrue("Expected file.encoding=ISO-8859-1", "ISO-8859-1".equals(System.getProperty("file.encoding")));
171         Charset.defaultCharset();
172         String outputDir = System.getProperty("output.dir", "target/anttasks");
173         String selftestOutput = System.getProperty("report.file", outputDir + "/selftest.report");
174         buildRule.executeTarget("testISO88591WithReportFile");
175         DocumentBuilder db = DocumentBuilderFactory.newInstance().newDocumentBuilder();
176         boolean documentParsed = false;
177         try (FileInputStream fis = new FileInputStream(selftestOutput)) {
178             Document doc = db.parse(fis);
179             boolean byteSequencePresent = doc.getElementsByTagName("header-sample")
180                     .item(0)
181                     .getTextContent()
182                     .contains("\u00E4\u00F6\u00FC\u00C4\u00D6\u00DC\u00DF");
183             assertTrue("Report should contain test umlauts", byteSequencePresent);
184             documentParsed = true;
185         } catch (Exception ex) {
186             documentParsed = false;
187         }
188         assertTrue("Report file could not be parsed as XML", documentParsed);
189     }
